00263 Bits CPU_Core_Dyn_X86_Run(void) {
00264     /* Dynamic core is NOT compatible with the way page faults
00265      * in the guest are handled in this emulator. Do not use
00266      * dynamic core if paging is enabled. Do not comment this
00267      * out, even if it happens to work for a minute, a half
00268      * hour, a day, because it will turn around and cause
00269      * Windows 95 to crash when you've become most comfortable
00270      * with the idea that it works. This code cannot handle
00271      * the sudden context switch of a page fault and it never
00272      * will. Don't do it. You have been warned. */
00273     if (paging.enabled) {
00274         if (paging_warning) {
00275             LOG_MSG("Dynamic core warning: The guest OS/Application has just switched on 80386 paging, which is not supported by the dynamic core. The normal core will be used until paging is switched off again.");
00276             paging_warning = false;
00277         }

00279         return CPU_Core_Normal_Run();
00280     }

00285 #if C_DEBUG
00286 #if C_HEAVY_DEBUG
00287                 if (DEBUG_HeavyIsBreakpoint()) return debugCallback;
00288 #endif
00289 #endif
00290         CodePageHandler * chandler=0;
00291         if (GCC_UNLIKELY(MakeCodePage(ip_point,chandler))) {
00292                 CPU_Exception(cpu.exception.which,cpu.exception.error);
00293                 goto restart_core;
00294         }
00295         if (!chandler) {
00296                 if (dyn_dh_fpu.state_used) DH_FPU_SAVE_REINIT
00297                 return CPU_Core_Normal_Run();
00298         }
00299         /* Find correct Dynamic Block to run */
00300         CacheBlock * block=chandler->FindCacheBlock(ip_point&4095);
00301         if (!block) {
00302                 if (!chandler->invalidation_map || (chandler->invalidation_map[ip_point&4095]<4)) {
00303                         block=CreateCacheBlock(chandler,ip_point,dynamic_core_cache_block_size);
00304                 } else {
00305                         Bitu old_cycles=CPU_Cycles;
00306                         CPU_Cycles=1;
00307                         Bits nc_retcode=CPU_Core_Normal_Run();
00308                         if (dyn_dh_fpu.state_used) DH_FPU_SAVE_REINIT
00309                         if (!nc_retcode) {
00310                                 CPU_Cycles=old_cycles-1;
00311                                 goto restart_core;
00312                         }
00313                         CPU_CycleLeft+=old_cycles;
00314                         return nc_retcode; 
00315                 }
00316         }

00319         cache.block.running=0;
00320         BlockReturn ret=gen_runcode(block->cache.start);
00321         cycle_count += CPU_CyclesOld - CPU_Cycles;
00322         switch (ret) {
00323         case BR_Iret:
00324 #if C_DEBUG
00325 #if C_HEAVY_DEBUG
00326                 if (DEBUG_HeavyIsBreakpoint()) {

00334                                 if (dyn_dh_fpu.state_used) DH_FPU_SAVE_REINIT
00335                                 return CBRET_NONE;
00336                         }
00337                         goto restart_core;
00338                 }
00339                 cpudecoder=CPU_Core_Dyn_X86_Trap_Run;
00340                 if (!dyn_dh_fpu.state_used) return CBRET_NONE;
00341                 DH_FPU_SAVE_REINIT
00342                 return CBRET_NONE;
00343         case BR_Normal:
00344                 /* Maybe check if we staying in the same page? */
00345 #if C_DEBUG
00346 #if C_HEAVY_DEBUG
00347                 if (DEBUG_HeavyIsBreakpoint()) return debugCallback;
00348 #endif
00349 #endif
00350                 goto restart_core;
00351         case BR_Cycles:
00352 #if C_DEBUG
00353 #if C_HEAVY_DEBUG                       
00354                 if (DEBUG_HeavyIsBreakpoint()) return debugCallback;
00355 #endif
00356 #endif

00358                 DH_FPU_SAVE_REINIT
00359                 return CBRET_NONE;
00360         case BR_CallBack:
00361                 if (!dyn_dh_fpu.state_used) return core_dyn.callback;
00362                 DH_FPU_SAVE_REINIT
00363                 return core_dyn.callback;
00364         case BR_SMCBlock:
00365 //              LOG_MSG("selfmodification of running block at %x:%x",SegValue(cs),reg_eip);
00366                 cpu.exception.which=0;
00367                 // fallthrough, let the normal core handle the block-modifying instruction
00368         case BR_Opcode:
00369                 CPU_CycleLeft+=CPU_Cycles;
00370                 CPU_Cycles=1;
00371                 if (dyn_dh_fpu.state_used) DH_FPU_SAVE_REINIT

00381         case BR_Link2:
00382                 {
00383                         Bitu temp_ip=SegPhys(cs)+reg_eip;
00384                         CodePageHandler * temp_handler=(CodePageHandler *)get_tlb_readhandler(temp_ip);

00389                                 goto run_block;
00390                         }
00391                 }

00393         }
00394         if (dyn_dh_fpu.state_used) DH_FPU_SAVE_REINIT
00395         return CBRET_NONE;
00396 }
