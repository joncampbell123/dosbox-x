00052                 if (!_dynreg) return;

00071                 if (GCC_UNLIKELY(!((Bitu)dynreg))) return;

00073                         Save();

00087 static BlockReturn gen_runcode(Bit8u * code) {
00088         BlockReturn retval;
00089 #if defined (_MSC_VER)
00090         __asm {
00091 /* Prepare the flags */
00092                 mov             eax,[code]
00093                 push    ebx
00094                 push    ebp
00095                 push    esi
00096                 push    edi
00097                 mov             ebx,[reg_flags]
00098                 and             ebx,FMASK_TEST
00099                 push    offset(return_address)
00100                 push    ebx
00101                 jmp             eax
00102 /* Restore the flags */
00103 return_address:
00104                 /*      return here with flags in ecx */
00105                 and             dword ptr [reg_flags],~FMASK_TEST
00106                 and             ecx,FMASK_TEST
00107                 or              [reg_flags],ecx
00108                 pop             edi
00109                 pop             esi
00110                 pop             ebp
00111                 pop             ebx
00112                 mov             [retval],eax
00113         }
00114 #elif defined (MACOSX)
00115         register Bit32u tempflags=reg_flags & FMASK_TEST;
00116         __asm__ volatile (
00117                 "pushl %%ebx                                            \n"
00118                 "pushl %%ebp                                            \n"
00119                 "pushl $(run_return_adress)                     \n"
00120                 "pushl  %2                                                      \n"
00121                 "jmp  *%3                                                       \n"
00122                 "run_return_adress:                                     \n"
00123                 "popl %%ebp                                                     \n"
00124                 "popl %%ebx                                                     \n"
00125                 :"=a" (retval), "=c" (tempflags)
00126                 :"r" (tempflags),"r" (code)
00127                 :"%edx","%edi","%esi","cc","memory"
00128         );
00129         reg_flags=(reg_flags & ~FMASK_TEST) | (tempflags & FMASK_TEST);
00130 #else
00131         register Bit32u tempflags=reg_flags & FMASK_TEST;
00132         __asm__ volatile (
00133                 "pushl $(run_return_adress)                                     \n"
00134                 "pushl  %2                                                      \n"
00135                 "jmp  *%3                                                       \n"
00136                 "run_return_adress:                                             \n"
00137                 :"=a" (retval), "=c" (tempflags)
00138                 :"r" (tempflags),"r" (code)
00139                 :"%edx","%ebx","%edi","%esi","%ebp","cc","memory"

00146 static GenReg * FindDynReg(DynReg * dynreg,bool stale=false) {
00147         x86gen.last_used++;
00148         if (dynreg->genreg) {
00149                 dynreg->genreg->last_used=x86gen.last_used;
00150                 return dynreg->genreg;
00151         }
00152         /* Find best match for selected global reg */
00153         Bits i;
00154         Bits first_used,first_index;
00155         first_used=-1;
00156         if (dynreg->flags & DYNFLG_HAS8) {
00157                 /* Has to be eax,ebx,ecx,edx */
00158                 for (i=first_index=0;i<=X86_REG_EBX;i++) {
00159                         GenReg * genreg=x86gen.regs[i];
00160                         if (genreg->notusable) continue;
00161                         if (!(genreg->dynreg)) {
00162                                 genreg->Load(dynreg,stale);
00163                                 return genreg;
00164                         }
00165                         if (genreg->last_used<(Bitu)first_used) {
00166                                 first_used=genreg->last_used;
00167                                 first_index=i;
00168                         }
00169                 }
00170         } else {
00171                 for (i=first_index=X86_REGS-1;i>=0;i--) {
00172                         GenReg * genreg=x86gen.regs[i];
00173                         if (genreg->notusable) continue;
00174                         if (!(genreg->dynreg)) {
00175                                 genreg->Load(dynreg,stale);
00176                                 return genreg;
00177                         }
00178                         if (genreg->last_used<(Bitu)first_used) {
00179                                 first_used=genreg->last_used;
00180                                 first_index=i;
00181                         }
00182                 }
00183         }
00184         /* No free register found use earliest assigned one */
00185         GenReg * newreg=x86gen.regs[first_index];
00186         newreg->Load(dynreg,stale);
00187         return newreg;
00188 }

00735 static void gen_call_function(void * func,char const* ops,...) {
00736         Bits paramcount=0;
00737         bool release_flags=false;
00738         struct ParamInfo {
00739                 const char * line;
00740                 Bitu value;
00741         } pinfo[32];
00742         ParamInfo * retparam=0;
00743         /* Clear the EAX Genreg for usage */
00744         x86gen.regs[X86_REG_EAX]->Clear();
00745         x86gen.regs[X86_REG_EAX]->notusable=true;
00746         /* Save the flags */
00747         if (GCC_UNLIKELY(!skip_flags)) gen_protectflags();
00748         /* Scan for the amount of params */
00749         if (ops) {
00750                 va_list params;
00751                 va_start(params,ops);
00752 #if defined (MACOSX)
00753                 Bitu stack_used=0;
00754                 bool free_flags=false;
00755 #endif
00756                 Bits pindex=0;
00757                 while (*ops) {
00758                         if (*ops=='%') {
00759                                 pinfo[pindex].line=ops+1;
00760                                 pinfo[pindex].value=va_arg(params,Bitu);
00761 #if defined (MACOSX)
00762                                 const char * scan=pinfo[pindex].line;
00763                                 if ((*scan=='I') || (*scan=='D')) stack_used+=4;
00764                                 else if (*scan=='F') free_flags=true;
00765 #endif
00766                                 pindex++;
00767                         }
00768                         ops++;
00769                 }
00770 
00771 #if defined (MACOSX)
00772                 /* align stack */
00773                 stack_used+=4;                  // saving esp on stack as well
00774 
00775                 cache_addw(0xc48b);             // mov eax,esp
00776                 cache_addb(0x2d);               // sub eax,stack_used
00777                 cache_addd(stack_used);
00778                 cache_addw(0xe083);             // and eax,0xfffffff0
00779                 cache_addb(0xf0);
00780                 cache_addb(0x05);               // sub eax,stack_used
00781                 cache_addd(stack_used);
00782                 cache_addb(0x94);               // xchg eax,esp
00783                 if (free_flags) {
00784                         cache_addw(0xc083);     // add eax,4
00785                         cache_addb(0x04);
00786                 }
00787                 cache_addb(0x50);               // push eax (==old esp)
00788 #endif
00789 
00790                 paramcount=0;
00791                 while (pindex) {
00792                         pindex--;
00793                         const char * scan=pinfo[pindex].line;
00794                         switch (*scan++) {
00795                         case 'I':                               /* immediate value */
00796                                 paramcount++;
00797                                 cache_addb(0x68);                       //Push immediate
00798                                 cache_addd(pinfo[pindex].value);        //Push value
00799                                 break;
00800                         case 'D':                               /* Dynamic register */
00801                                 {
00802                                         bool release=false;
00803                                         paramcount++;
00804                                         DynReg * dynreg=(DynReg *)pinfo[pindex].value;
00805                                         GenReg * genreg=FindDynReg(dynreg);
00806                                         scanagain:
00807                                         switch (*scan++) {
00808                                         case 'd':
00809                                                 cache_addb(0x50+genreg->index);         //Push reg
00810                                                 break;
00811                                         case 'w':
00812                                                 cache_addw(0xb70f);                                     //MOVZX EAX,reg
00813                                                 cache_addb(0xc0+genreg->index);
00814                                                 cache_addb(0x50);                                       //Push EAX
00815                                                 break;
00816                                         case 'l':
00817                                                 cache_addw(0xb60f);                                     //MOVZX EAX,reg[0]
00818                                                 cache_addb(0xc0+genreg->index);
00819                                                 cache_addb(0x50);                                       //Push EAX
00820                                                 break;
00821                                         case 'h':
00822                                                 cache_addw(0xb60f);                                     //MOVZX EAX,reg[1]
00823                                                 cache_addb(0xc4+genreg->index);
00824                                                 cache_addb(0x50);                                       //Push EAX
00825                                                 break;
00826                                         case 'r':                                                               /* release the reg afterwards */
00827                                                 release=true;
00828                                                 goto scanagain;
00829                                         default:
00830                                                 IllegalOption("gen_call_function param:DREG");
00831                                         }
00832                                         if (release) gen_releasereg(dynreg);
00833                                 }
00834                                 break;
00835                         case 'R':                               /* Dynamic register to get the return value */
00836                                 retparam =&pinfo[pindex];
00837                                 pinfo[pindex].line=scan;
00838                                 break;
00839                         case 'F':                               /* Release flags from stack */
00840                                 release_flags=true;
00841                                 break;
00842                         default:
00843                                 IllegalOption("gen_call_function unknown param");
00844                         }
00845                 }
00846 #if defined (MACOSX)
00847                 if (free_flags) release_flags=false;
00848         } else {
00849                 /* align stack */
00850                 Bit32u stack_used=8;    // saving esp and return address on the stack
00851 
00852                 cache_addw(0xc48b);             // mov eax,esp
00853                 cache_addb(0x2d);               // sub eax,stack_used
00854                 cache_addd(stack_used);
00855                 cache_addw(0xe083);             // and eax,0xfffffff0
00856                 cache_addb(0xf0);
00857                 cache_addb(0x05);               // sub eax,stack_used
00858                 cache_addd(stack_used);
00859                 cache_addb(0x94);               // xchg eax,esp
00860                 cache_addb(0x50);               // push esp (==old esp)
00861 #endif
00862         }
00863 
00864         /* Clear some unprotected registers */
00865         x86gen.regs[X86_REG_ECX]->Clear();
00866         x86gen.regs[X86_REG_EDX]->Clear();
00867         /* Do the actual call to the procedure */
00868         cache_addb(0xe8);
00869         cache_addd((uintptr_t)func - (uintptr_t)cache.pos - (uintptr_t)4);
00870         /* Restore the params of the stack */
00871         if (paramcount) {
00872                 cache_addw(0xc483);                             //add ESP,imm byte
00873                 cache_addb(paramcount*4+(release_flags?4:0));
00874         } else if (release_flags) {
00875                 cache_addw(0xc483);                             //add ESP,imm byte
00876                 cache_addb(4);
00877         }
00878         /* Save the return value in correct register */
00879         if (retparam) {
00880                 DynReg * dynreg=(DynReg *)retparam->value;
00881                 GenReg * genreg=FindDynReg(dynreg);
00882                 if (genreg->index)              // test for (e)ax/al
00883                 switch (*retparam->line) {
00884                 case 'd':
00885                         cache_addw(0xc08b+(genreg->index <<(8+3)));     //mov reg,eax
00886                         break;
00887                 case 'w':
00888                         cache_addb(0x66);                                                       
00889                         cache_addw(0xc08b+(genreg->index <<(8+3)));     //mov reg,eax
00890                         break;
00891                 case 'l':
00892                         cache_addw(0xc08a+(genreg->index <<(8+3)));     //mov reg,eax
00893                         break;
00894                 case 'h':
00895                         cache_addw(0xc08a+((genreg->index+4) <<(8+3))); //mov reg,eax
00896                         break;
00897                 }
00898                 dynreg->flags|=DYNFLG_CHANGED;
00899         }
00900         /* Restore EAX registers to be used again */
00901         x86gen.regs[X86_REG_EAX]->notusable=false;
00902 
00903 #if defined (MACOSX)
00904         /* restore stack */
00905         cache_addb(0x5c);       // pop esp
00906 #endif
00907 }
