00064     }
00065     else if (pc98_gdc_vramop & (1 << VOPBIT_ANALOG)) {

00070             vga.dac.rgb[i].red   = dac_4to6(pc98_pal_analog[(3*i) + 1]&0xF); /* re-use VGA DAC */
00071             vga.dac.rgb[i].blue  = dac_4to6(pc98_pal_analog[(3*i) + 2]&0xF); /* re-use VGA DAC */
00072             VGA_DAC_UpdateColor(i);
00073         }
00074     }
00075     else {
00076         vga_8bit_dac = false;
