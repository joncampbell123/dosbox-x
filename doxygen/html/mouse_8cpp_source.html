00224 void DoPS2Callback(Bit16u data, Bit16s mouseX, Bit16s mouseY) {
00225         if (useps2callback && ps2cbseg != 0 && ps2cbofs != 0) {
00226                 Bit16u mdat = (data & 0x03) | 0x08;
00227                 Bit16s xdiff = mouseX-oldmouseX;
00228                 Bit16s ydiff = oldmouseY-mouseY;
00229                 oldmouseX = mouseX;
00230                 oldmouseY = mouseY;
00231                 if ((xdiff>0xff) || (xdiff<-0xff)) mdat |= 0x40;                // x overflow
00232                 if ((ydiff>0xff) || (ydiff<-0xff)) mdat |= 0x80;                // y overflow
00233                 xdiff %= 256;
00234                 ydiff %= 256;
00235                 if (xdiff<0) {
00236                         xdiff = (0x100+xdiff);
00237                         mdat |= 0x10;
00238                 }
00239                 if (ydiff<0) {
00240                         ydiff = (0x100+ydiff);
00241                         mdat |= 0x20;
00242                 }
00243                 if (ps2_callback_save_regs) {
00244                         CPU_Push16(reg_ax);CPU_Push16(reg_cx);CPU_Push16(reg_dx);CPU_Push16(reg_bx);
00245                         CPU_Push16(reg_bp);CPU_Push16(reg_si);CPU_Push16(reg_di);
00246                         CPU_Push16(SegValue(ds)); CPU_Push16(SegValue(es));
00247                 }
00248                 CPU_Push16((Bit16u)mdat); 
00249                 CPU_Push16((Bit16u)(xdiff % 256)); 
00250                 CPU_Push16((Bit16u)(ydiff % 256)); 
00251                 CPU_Push16((Bit16u)0); 
00252                 CPU_Push16(RealSeg(ps2_callback));
00253                 CPU_Push16(RealOff(ps2_callback));
00254                 SegSet16(cs, ps2cbseg);
00255                 reg_ip = ps2cbofs;
00256         }
00257 }

00301 INLINE void Mouse_AddEvent(Bit8u type) {
00302         if (mouse.events<QUEUE_SIZE) {
00303                 if (mouse.events>0) {
00304                         /* Skip duplicate events */
00305                         if (type==MOUSE_HAS_MOVED) return;
00306                         /* Always put the newest element in the front as that the events are 
00307                          * handled backwards (prevents doubleclicks while moving)
00308                          */
00309                         for(Bitu i = mouse.events ; i ; i--)
00310                                 mouse.event_queue[i] = mouse.event_queue[i-1];
00311                 }
00312                 mouse.event_queue[0].type=type;
00313                 mouse.event_queue[0].buttons=mouse.buttons;
00314                 mouse.events++;
00315         }
00316         if (!mouse.timer_in_progress) {
00317                 mouse.timer_in_progress = true;
00318                 PIC_AddEvent(MOUSE_Limit_Events,MOUSE_DELAY);
00319 
00320         if (!IS_PC98_ARCH || (IS_PC98_ARCH && p7fd8_8255_mouse_int_enable))
00321             PIC_ActivateIRQ(MOUSE_IRQ);

01187 static Bitu INT74_Handler(void) {
01188         if (mouse.events>0) {
01189                 mouse.events--;

01191                 /* INT 33h emulation: HERE within the IRQ 12 handler is the appropriate place to
01192                  * redraw the cursor. OSes like Windows 3.1 expect real-mode code to do it in
01193                  * response to IRQ 12, not "out of the blue" from the SDL event handler like
01194                  * the original DOSBox code did it. Doing this allows the INT 33h emulation
01195                  * to draw the cursor while not causing Windows 3.1 to crash or behave
01196                  * erratically. */
01197                 if (en_int33) DrawCursor();

01199                 /* Check for an active Interrupt Handler that will get called */
01200                 if (mouse.sub_mask & mouse.event_queue[mouse.events].type) {
01201                         reg_ax=mouse.event_queue[mouse.events].type;
01202                         reg_bx=mouse.event_queue[mouse.events].buttons;
01203                         reg_cx=POS_X;

01205                         reg_si=static_cast<Bit16s>(mouse.mickey_x);
01206                         reg_di=static_cast<Bit16s>(mouse.mickey_y);
01207                         CPU_Push16(RealSeg(CALLBACK_RealPointer(int74_ret_callback)));
01208                         CPU_Push16(RealOff(CALLBACK_RealPointer(int74_ret_callback)));

01210                         reg_ip = mouse.sub_ofs;
01211                         if(mouse.in_UIR) LOG(LOG_MOUSE,LOG_ERROR)("Already in UIR!");

01214                         CPU_Push16(RealSeg(CALLBACK_RealPointer(int74_ret_callback)));
01215                         CPU_Push16(RealOff(CALLBACK_RealPointer(int74_ret_callback)));
01216                         DoPS2Callback(mouse.event_queue[mouse.events].buttons, static_cast<Bit16s>(mouse.ps2x), static_cast<Bit16s>(mouse.ps2y));

01218                         SegSet16(cs, RealSeg(CALLBACK_RealPointer(int74_ret_callback)));
01219                         reg_ip = RealOff(CALLBACK_RealPointer(int74_ret_callback));
