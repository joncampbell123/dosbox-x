01873     void writecmd(const unsigned char b) { /* write to command register */
01874         if (state == MODE_STATE) {
01875             mode_byte = b;
01876 
01877             if ((b&3) != 0) {
01878                 /* bit[7:6] = number of stop bits  (0=invalid 1=1-bit 2=1.5-bit 3=2-bit)
01879                  * bit[5:5] = even/odd parity      (1=even 0=odd)
01880                  * bit[4:4] = parity enable        (1=enable 0=disable)
01881                  * bit[3:2] = character length     (0=5  1=6  2=7  3=8)
01882                  * bit[1:0] = baud rate factor     (0=sync mode   1=1X   2=16X   3=64X)
01883                  *
01884                  * note that "baud rate factor" means how much to divide the baud rate clock to determine
01885                  * the bit rate that bits are transmitted. Typical PC-98 programming practice is to set
01886                  * the baud rate clock fed to the chip at 16X the baud rate and then specify 16X baud rate factor. */
01887                 /* async mode */
01888                 state = COMMAND_STATE;
01889 
01890                 /* keyboard must operate at 19200 baud 8 bits odd parity 16X baud rate factor */
01891                 valid_state = (b == 0x5E); /* bit[7:0] = 01 0 1 11 10 */
01892                                            /*            |  | | |  |  */
01893                                            /*            |  | | |  +---- 16X baud rate factor */
01894                                            /*            |  | | +------- 8 bits per character */
01895                                            /*            |  | +--------- parity enable */
01896                                            /*            |  +----------- odd parity */
01897                                            /*            +-------------- 1 stop bit */
01898             }
01899             else {
01900                 /* bit[7:7] = single character sync(1=single  0=double)
01901                  * bit[6:6] = external sync detect (1=syndet is an input   0=syndet is an output)
01902                  * bit[5:5] = even/odd parity      (1=even 0=odd)
01903                  * bit[4:4] = parity enable        (1=enable 0=disable)
01904                  * bit[3:2] = character length     (0=5  1=6  2=7  3=8)
01905                  * bit[1:0] = baud rate factor     (0=sync mode)
01906                  *
01907                  * I don't think anything uses the keyboard in this manner, therefore, not supported in this emulation. */
01908                 LOG_MSG("8251 keyboard warning: Mode byte synchronous mode not supported");
01909                 state = SYNC_CHAR1;
01910                 valid_state = false;
01911             }
01912         }
01913         else if (state == COMMAND_STATE) {
01914             /* bit[7:7] = Enter hunt mode (not used here)
01915              * bit[6:6] = internal reset (8251 resets, prepares to accept mode byte)
01916              * bit[5:5] = RTS inhibit (1=force RTS to zero, else RTS reflects RxRDY state of the chip)
01917              * bit[4:4] = error reset
01918              * bit[3:3] = send break character (0=normal  1=force TxD low). On PC-98 keyboard this is wired to reset pin of the keyboard CPU.
01919              * bit[2:2] = receive enable
01920              * bit[1:1] = DTR inhibit (1=force DTR to zero). Connected to PC-98 RTY pin.
01921              * bit[0:0] = transmit enable */
01922             if (b & 0x40) {
01923                 /* internal reset, returns 8251 to mode state */
01924                 state = MODE_STATE;
01925             }
01926 
01927             /* TODO: Does the 8251 take any other bits if bit 6 was set to reset the 8251? */
01928             keyboard_reset = !!(b & 0x08);
01929             rx_enable = !!(b & 0x04);
01930             tx_enable = !!(b & 0x01);
01931         }
01932     }
