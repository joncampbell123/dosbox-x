00334     if(!installed) {
00335         if (pagemask == 0 || (pagemask & ~0xFFFFFFFU)) {
00336             LOG(LOG_MISC,LOG_ERROR)("MEM_CalloutObject::Install: Page mask %x is invalid",(unsigned int)pagemask);
00337             return;
00338         }
00339 
00340         /* we need a mask for the distance between aliases of the port, and the range of I/O ports. */
00341         /* only the low part of the mask where bits are zero, not the upper.
00342          * This loop is the reason portmask cannot be ~0 else it would become an infinite loop.
00343          * This also serves to check that the mask is a proper combination of ISA masking and
00344          * I/O port range (example: IOMASK_ISA_10BIT & (~0x3) == 0x3FF & 0xFFFFFFFC = 0x3FC for a device with 4 I/O ports).
00345          * A proper mask has (from MSB to LSB):
00346          *   - zero or more 0 bits from MSB
00347          *   - 1 or more 1 bits in the middle
00348          *   - zero or more 0 bits to LSB */
00349         {
00350             Bitu m = 1;
00351             Bitu test;
00352 
00353             /* compute range mask from zero bits at LSB */
00354             range_mask = 0;
00355             test = pagemask ^ 0xFFFFFFFU;
00356             while ((test & m) == m) {
00357                 range_mask = m;
00358                 m = (m << 1) + 1;
00359             }
00360 
00361             /* DEBUG */
00362             if ((pagemask & range_mask) != 0 ||
00363                 ((range_mask + 1) & range_mask) != 0/* should be a mask, therefore AND by itself + 1 should be zero (think (0xF + 1) & 0xF = 0x10 & 0xF = 0x0) */) {
00364                 LOG(LOG_MISC,LOG_ERROR)("MEM_CalloutObject::Install: pagemask(%x) & range_mask(%x) != 0 (%x). You found a corner case that broke this code, fix it.",
00365                     (unsigned int)pagemask,
00366                     (unsigned int)range_mask,
00367                     (unsigned int)(pagemask & range_mask));
00368                 return;
00369             }
00370 
00371             /* compute alias mask from middle 1 bits */
00372             alias_mask = range_mask;
00373             test = pagemask + range_mask; /* will break if portmask & range_mask != 0 */
00374             while ((test & m) == m) {
00375                 alias_mask = m;
00376                 m = (m << 1) + 1;
00377             }
00378 
00379             /* any bits after that should be zero. */
00380             /* confirm this by XORing portmask by (alias_mask ^ range_mask). */
00381             /* we already confirmed that portmask & range_mask == 0. */
00382             /* Example:
00383              *
00384              *    Sound Blaster at port 220-22Fh with 10-bit ISA decode would be 0x03F0 therefore:
00385              *      portmask =    0x03F0
00386              *      range_mask =  0x000F
00387              *      alias_mask =  0x03FF
00388              *
00389              *      portmask ^ range_mask = 0x3FF
00390              *      portmask ^ range_mask ^ alias_mask = 0x0000
00391              *
00392              * Example of invalid portmask 0x13F0:
00393              *      portmask =    0x13F0
00394              *      range_mask =  0x000F
00395              *      alias_mask =  0x03FF
00396              *
00397              *      portmask ^ range_mask = 0x13FF
00398              *      portmask ^ range_mask ^ alias_mask = 0x1000 */
00399             if ((pagemask ^ range_mask ^ alias_mask) != 0 ||
00400                 ((alias_mask + 1) & alias_mask) != 0/* should be a mask, therefore AND by itself + 1 should be zero */) {
00401                 LOG(LOG_MISC,LOG_ERROR)("MEM_CalloutObject::Install: pagemask(%x) ^ range_mask(%x) ^ alias_mask(%x) != 0 (%x). Invalid portmask.",
00402                     (unsigned int)pagemask,
00403                     (unsigned int)range_mask,
00404                     (unsigned int)alias_mask,
00405                     (unsigned int)(pagemask ^ range_mask ^ alias_mask));
00406                 return;
00407             }
00408 
00409             if (page & range_mask) {
00410                 LOG(LOG_MISC,LOG_ERROR)("MEM_CalloutObject::Install: page %x and page mask %x not aligned (range_mask %x)",
00411                     (unsigned int)page,(unsigned int)pagemask,(unsigned int)range_mask);
00412                 return;
00413             }
00414         }
00415 
00416         installed=true;
00417         m_base=page;
00418         mem_mask=pagemask;
00419         m_handler=handler;
00420 
00421         /* add this object to the callout array.
00422          * don't register any I/O handlers. those will be registered during the "slow path"
00423          * callout process when the CPU goes to access them. to encourage that to happen,
00424          * we invalidate the I/O ranges */
00425         LOG(LOG_MISC,LOG_DEBUG)("MEM_CalloutObject::Install added device with page=0x%x mem_mask=0x%x rangemask=0x%x aliasmask=0x%x",
00426             (unsigned int)page,(unsigned int)mem_mask,(unsigned int)range_mask,(unsigned int)alias_mask);
00427 
00428         InvalidateCachedHandlers();
00429     }

00438 /* callers maintain a handle to it.
00439  * if they need to touch it, they get a pointer, which they then have to put back.
00440  * The way DOSBox/DOSbox-X code handles MEM callbacks it's common to declare an MEM object,
00441  * call the install, but then never touch it again, so this should work fine.
00442  *
00443  * this allows us to maintain ready-made MEM callout objects to return quickly rather
00444  * than write more complicated code where the caller has to make an MEM_CalloutObject
00445  * and then call install and we have to add it's pointer to a list/vector/whatever.
00446  * It also avoids problems where if we have to resize the vector, the pointers become
00447  * invalid, because callers have only handles and they have to put all the pointers
00448  * back in order for us to resize the vector. */

00450     if (t < MEM_TYPE_MIN || t >= MEM_TYPE_MAX)
00451         return MEM_Callout_t_none;
00452 
00453     MEM_callout_vector &vec = MEM_callouts[t - MEM_TYPE_MIN];
00454 
00455 try_again:
00456     while (vec.alloc_from < vec.size()) {
00457         MEM_CalloutObject &obj = vec[vec.alloc_from];
00458 
00459         if (!obj.alloc) {
00460             obj.alloc = true;
00461             assert(obj.isInstalled() == false);
00462             return MEM_Callout_t_comb(t,vec.alloc_from++); /* make combination, then increment alloc_from */
00463         }
00464 
00465         vec.alloc_from++;
00466     }
00467 
00468     /* okay, double the size of the vector within reason.
00469      * if anyone has pointers out to our elements, then we cannot resize. vector::resize() may invalidate them. */
00470     if (vec.size() < 4096 && vec.getcounter == 0) {
00471         size_t nsz = vec.size() * 2;
00472 
00473         LOG(LOG_MISC,LOG_WARN)("MEM_AllocateCallout type %u expanding array to %u",(unsigned int)t,(unsigned int)nsz);
00474         vec.alloc_from = (unsigned int)vec.size(); /* allocate from end of old vector size */
00475         vec.resize(nsz);
00476         goto try_again;
00477     }
00478 
00479     LOG(LOG_MISC,LOG_WARN)("MEM_AllocateCallout type %u no free entries",(unsigned int)t);

00713     Bit8u const* read = reinterpret_cast<Bit8u const*>(data);

01312 void On_Software_CPU_Reset() {

01314 
01315     CPU_Exception_Level_Reset();
01316 
01317     if (custom_bios) {
01318         /* DO NOTHING */
01319         LOG_MSG("CPU RESET: Doing nothing, custom BIOS loaded");
01320 
01321         if (IS_PC98_ARCH)
01322             LOG_MSG("CPU RESET: SHUT0=%u SHUT1=%u",PC98_SHUT0,PC98_SHUT1);
01323         else
01324             LOG_MSG("CPU RESET: CMOS BYTE 0x%02x",CMOS_GetShutdownByte());
01325     }
01326     else if (IS_PC98_ARCH) {
01327         /* From Undocumented 9801, 9821 Volume 2:
01328          *
01329          * SHUT0 | SHUT1 | Meaning
01330          * -----------------------
01331          * 1       1       System reset (BIOS performs full reinitialization)
01332          * 1       0       Invalid (BIOS will show "SYSTEM SHUTDOWN" and stop)
01333          * 0       x       Continue program execution after CPU reset.
01334          *                    BIOS loads SS:SP from 0000:0404 and then executes RETF */
01335         if (PC98_SHUT0) {
01336             if (!PC98_SHUT1)
01337                 E_Exit("PC-98 invalid reset aka SYSTEM SHUTDOWN (SHUT0=1 SHUT1=0)");
01338         }
01339         else { // SHUT0=0 SHUT1=x
01340             void CPU_Snap_Back_To_Real_Mode();
01341             void CPU_Snap_Back_Forget();
01342 
01343             /* fake CPU reset */
01344             CPU_Snap_Back_To_Real_Mode();
01345             CPU_Snap_Back_Forget();
01346 
01347             /* following CPU reset, and coming from the BIOS, CPU registers are trashed */
01348             /* FIXME: VEM486.EXE appears to use this reset vector trick, then when regaining control,
01349              *        checks to see if DX is 0x00F0 just as it was when it issued the OUT DX,AL
01350              *        instruction. Why? If DX != 0x00F0 it writes whatever DX is to 0000:0486 and
01351              *        then proceeds anyway. */
01352             reg_eax = 0x2010000;
01353             reg_ebx = 0x2111;
01354             reg_ecx = 0;
01355             reg_edx = 0xABCD;
01356             reg_esi = 0;
01357             reg_edi = 0;
01358             reg_ebp = 0;
01359             reg_esp = 0x4F8;
01360             CPU_SetSegGeneral(ds,0x0040);
01361             CPU_SetSegGeneral(es,0x0000);
01362             CPU_SetSegGeneral(ss,0x0000);
01363 
01364             /* continue program execution after CPU reset */
01365             Bit16u reset_sp = mem_readw(0x404);
01366             Bit16u reset_ss = mem_readw(0x406);
01367 
01368             LOG_MSG("PC-98 reset and continue: SS:SP = %04x:%04x",reset_ss,reset_sp);
01369 
01370             reg_esp = reset_sp;
01371             CPU_SetSegGeneral(ss,reset_ss);
01372 
01373             Bit16u new_ip = CPU_Pop16();
01374             Bit16u new_cs = CPU_Pop16();
01375 
01376             reg_eip = new_ip;
01377             CPU_SetSegGeneral(cs,new_cs);
01378 
01379             LOG_MSG("PC-98 reset and continue: RETF to %04x:%04x",SegValue(cs),reg_ip);
01380 
01381             // DEBUG
01382 //            Bitu DEBUG_EnableDebugger(void);
01383   //          DEBUG_EnableDebugger();
01384 
01385             /* force execution change (FIXME: Is there a better way to do this?) */
01386             throw int(4);
01387             /* does not return */
01388         }
01389     }
01390     else {
01391         /* IBM reset vector or system reset by CMOS shutdown byte */
01392 
01393         /* software-initiated CPU reset. but the intent may not be to reset the system but merely
01394          * the CPU. check the CMOS shutdown byte */
01395         switch (c=CMOS_GetShutdownByte()) {
01396             case 0x05:  /* JMP double word pointer with EOI */
01397             case 0x0A:  /* JMP double word pointer without EOI */

01399                 return;
01400             case 0x09:  /* INT 15h block move return to real mode (to appease Windows 3.1 KRNL286.EXE and cputype=286, yuck) */
01401                 On_Software_286_int15_block_move_return(c);
01402                 return;
01403         }
01404     }
01405 
01406 #if C_DYNAMIC_X86
01407     /* this technique is NOT reliable when running the dynamic core! */
01408     if (cpudecoder == &CPU_Core_Dyn_X86_Run) {
01409         E_Exit("C++ exception method is not compatible with dynamic core when emulating reset, aborting");
01410         return;
01411     }
01412 #endif
01413 
01414     throw int(3);
01415     /* does not return */
01416 }
