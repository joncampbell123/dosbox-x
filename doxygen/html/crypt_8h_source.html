00047  */
00048 static int update_keys(unsigned long* pkeys,const unsigned long* pcrc_32_tab,int c)
00049 {
00050     (*(pkeys+0)) = CRC32((*(pkeys+0)), c);
00051     (*(pkeys+1)) += (*(pkeys+0)) & 0xff;
