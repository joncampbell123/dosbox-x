00139         /* Check to make sure that  there is at least one setup image */

00992 Bit8u imageDiskVFD::Write_Sector(Bit32u head,Bit32u cylinder,Bit32u sector,void * data) {
00993     unsigned long new_offset;
00994     unsigned char tmp[12];
00995     vfdentry *ent;
00996 
00997     LOG_MSG("VFD write sector: CHS %u/%u/%u",cylinder,head,sector);
00998 
00999     ent = findSector(head,cylinder,sector);
01000     if (ent == NULL) return 0x05;
01001     if (ent->getSectorSize() != sector_size) return 0x05;
01002 
01003     if (ent->hasSectorData()) {
01004         fseek(diskimg,ent->data_offset,SEEK_SET);
01005         if (ftell(diskimg) != ent->data_offset) return 0x05;
01006         if (fwrite(data,sector_size,1,diskimg) != 1) return 0x05;
01007         return 0;
01008     }
01009     else if (ent->hasFill()) {
01010         bool isfill = false;
01011 
01012         /* well, is the data provided one character repeated?
01013          * note the format cannot represent a fill byte of 0xFF */
01014         if (((unsigned char*)data)[0] != 0xFF) {
01015             unsigned int i=1;
01016 
01017             do {
01018                 if (((unsigned char*)data)[i] == ((unsigned char*)data)[0]) {
01019                     if ((++i) == sector_size) {
01020                         isfill = true;
01021                         break; // yes!
01022                     }
01023                 }
01024                 else {
01025                     break; // nope
01026                 }
01027             } while (1);
01028         }
01029 
01030         if (ent->entry_offset == 0) return 0x05;
01031 
01032         if (isfill) {
01033             fseek(diskimg,ent->entry_offset,SEEK_SET);
01034             if (ftell(diskimg) != ent->entry_offset) return 0x05;
01035             if (fread(tmp,12,1,diskimg) != 1) return 0x05;
01036 
01037             tmp[0x04] = ((unsigned char*)data)[0]; // change the fill byte
01038 
01039             LOG_MSG("VFD write: 'fill' sector changing fill byte to 0x%x",tmp[0x04]);
01040 
01041             fseek(diskimg,ent->entry_offset,SEEK_SET);
01042             if (ftell(diskimg) != ent->entry_offset) return 0x05;
01043             if (fwrite(tmp,12,1,diskimg) != 1) return 0x05;
01044         }
01045         else {
01046             fseek(diskimg,0,SEEK_END);
01047             new_offset = ftell(diskimg);
01048 
01049             /* we have to change it from a fill sector to an actual sector */
01050             LOG_MSG("VFD write: changing 'fill' sector to one with data (data at %lu)",(unsigned long)new_offset);
01051 
01052             fseek(diskimg,ent->entry_offset,SEEK_SET);
01053             if (ftell(diskimg) != ent->entry_offset) return 0x05;
01054             if (fread(tmp,12,1,diskimg) != 1) return 0x05;
01055 
01056             tmp[0x00] = ent->track;
01057             tmp[0x01] = ent->head;
01058             tmp[0x02] = ent->sector;
01059             tmp[0x03] = ent->sizebyte;
01060             tmp[0x04] = 0xFF; // no longer a fill byte
01061             tmp[0x05] = 0x00; // TODO ??
01062             tmp[0x06] = 0x00; // TODO ??
01063             tmp[0x07] = 0x00; // TODO ??
01064             *((uint32_t*)(tmp+8)) = new_offset;
01065             ent->fillbyte = 0xFF;
01066             ent->data_offset = (uint32_t)new_offset;
01067 
01068             fseek(diskimg,ent->entry_offset,SEEK_SET);
01069             if (ftell(diskimg) != ent->entry_offset) return 0x05;
01070             if (fwrite(tmp,12,1,diskimg) != 1) return 0x05;
01071 
01072             fseek(diskimg,ent->data_offset,SEEK_SET);
01073             if (ftell(diskimg) != ent->data_offset) return 0x05;
01074             if (fwrite(data,sector_size,1,diskimg) != 1) return 0x05;
01075         }
01076     }
01077 
01078         return 0x05;
01079 }
