00135                         if (sscanf(gl_verstr,"%d.%d", &major, &minor) != 2) major = 0;
00136                 } else {
00137                         if (sscanf(gl_verstr, "%d", &major) != 1) major = 0;
00138                 }
00139                 if (major > 0) {
00140                         ogl_ver = major*100;
00141                         if (minor >= 0) {
00142                                 if (minor < 10) ogl_ver += minor*10;
00143                                 else ogl_ver += minor;
