00075 static void getPixel(Bits x, Bits y, int &r, int &g, int &b, int shift)
00076 {
00077         if (x >= (Bits)render.src.width) x = (Bits)render.src.width-1;
00078         if (y >= (Bits)render.src.height) x = (Bits)render.src.height-1;
00079         if (x < 0) x = 0;
00080         if (y < 0) y = 0;
00081 
00082         Bit8u* src = (Bit8u *)&scalerSourceCache;
00083         Bit32u pixel;
00084         switch (render.scale.inMode) {
00085         case scalerMode8:
00086                 pixel = *(x+(Bit8u*)(src+y*render.scale.cachePitch));
00087                 r += render.pal.rgb[pixel].red >> shift;
00088                 g += render.pal.rgb[pixel].green >> shift;
00089                 b += render.pal.rgb[pixel].blue >> shift;
00090                 break;
00091         case scalerMode15:
00092                 pixel = *(x+(Bit16u*)(src+y*render.scale.cachePitch));
00093                 r += (pixel >> (7+shift)) & (0xf8 >> shift);
00094                 g += (pixel >> (2+shift)) & (0xf8 >> shift);
00095                 b += (pixel << (3-shift)) & (0xf8 >> shift);
00096                 break;
00097         case scalerMode16:
00098                 pixel = *(x+(Bit16u*)(src+y*render.scale.cachePitch));
00099                 r += (pixel >> (8+shift)) & (0xf8 >> shift);
00100                 g += (pixel >> (3+shift)) & (0xfc >> shift);
00101                 b += (pixel << (3-shift)) & (0xf8 >> shift);
00102                 break;
00103         case scalerMode32:
00104                 pixel = *(x+(Bit32u*)(src+y*render.scale.cachePitch));
00105                 r += (pixel >> (16+shift)) & (0xff >> shift);
00106                 g += (pixel >> (8+shift))  & (0xff >> shift);
00107                 b += (pixel >> shift)      & (0xff >> shift);
00108                 break;
00109         }
00110 }

00293 template<typename T> inline std::string stringify(const T& x, std::ios_base& ( *pf )(std::ios_base&) = NULL) {
00294         std::ostringstream o;
00295         if (pf) o << pf;
