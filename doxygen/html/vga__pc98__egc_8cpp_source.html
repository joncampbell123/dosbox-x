00069 void pc98_egc4a0_write(Bitu port,Bitu val,Bitu iolen) {
00070     /* Neko Project II suggests the I/O ports disappear when not in EGC mode.
00071      * Is that true? */
00072     if (!(pc98_gdc_vramop & (1 << VOPBIT_EGC))) {
00073 //        LOG_MSG("EGC 4A0 write port 0x%x when EGC not enabled",(unsigned int)port);
00074         return;
00075     }
00076 
00077     /* assume: (port & 1) == 0 [even] and iolen == 2 */
00078     switch (port & 0x0E) {
00079         case 0x0: /* 0x4A0 */
00080             /* bits [15:8] = 0xFF
00081              * bits [7:0] = enable writing to plane (NTS: only bits 3-0 have meaning in 16-color mode).
00082              * as far as I can tell, bits [7:0] correspond to the same enable bits as port 0x7C [3:0] */
00083             pc98_egc_access = val & 0xFF;
00084             break;
00085         case 0x2: /* 0x4A2 */
00086             /* bits [15:15] = 0
00087              * bits [14:13] = foreground, background color
00088              *    11 = invalid
00089              *    10 = foreground color
00090              *    01 = background color
00091              *    00 = pattern register
00092              * bits [12:12] = 0
00093              * bits [11:8] = lead plane
00094              *    0111 = VRAM plane #7
00095              *    0110 = VRAM plane #6
00096              *    0101 = VRAM plane #5
00097              *    0100 = VRAM plane #4
00098              *    0011 = VRAM plane #3
00099              *    0010 = VRAM plane #2
00100              *    0001 = VRAM plane #1
00101              *    0000 = VRAM plane #0
00102              * bits [7:0] = unused (0xFF) */
00103             pc98_egc_fgc = (val >> 13) & 3;
00104             pc98_egc_lead_plane = (val >> 8) & 15;
00105             break;
00106         case 0x4: /* 0x4A4 */
00107             /* bits [15:14] = 0 (unused)
00108              * bits [13:13] = 0=compare lead plane  1=don't
00109              * bits [12:11] = light source
00110              *    11 = invalid
00111              *    10 = write the contents of the palette register
00112              *    01 = write the result of the raster operation
00113              *    00 = write CPU data
00114              * bits [10:10] = read source
00115              *    1 = shifter input is CPU write data
00116              *    0 = shifter input is VRAM data
00117              * bits [9:8] = register load
00118              *    11 = invalid
00119              *    10 = load VRAM data before writing on VRAM write
00120              *    01 = load VRAM data into pattern/tile register on VRAM read
00121              *    00 = Do not change pattern/tile register
00122              * bits [7:0] = ROP
00123              *    shifter:       11110000
00124              *    destination:   11001100
00125              *    pattern reg:   10101010
00126              *
00127              *    examples:
00128              *    11110000 = VRAM transfer
00129              *    00001111 = VRAM reverse transfer
00130              *    11001100 = NOP
00131              *    00110011 = VRAM inversion
00132              *    11111111 = VRAM fill
00133              *    00000000 = VRAM erase
00134              *    10101010 = Pattern fill
00135              *    01010101 = Pattern reversal fill */
00136             pc98_egc_compare_lead = ((val >> 13) & 1) ^ 1;
00137             pc98_egc_lightsource = (val >> 11) & 3;
00138             pc98_egc_shiftinput = (val >> 10) & 1;
00139             pc98_egc_regload = (val >> 8) & 3;
00140             pc98_egc_rop = (val & 0xFF);
00141             break;
00142         case 0x6: /* 0x4A6 */
00143             /* If FGC = 0 and BGC = 0:
00144              *   bits [15:0] = 0
00145              * If FGC = 1 or BGC = 1:
00146              *   bits [15:8] = 0
00147              *   bits [7:0] = foreground color (all 8 bits used in 256-color mode) */
00148             pc98_egc_foreground_color = val;
00149             pc98_egc_fgcm[0].w = (val & 1) ? 0xFFFF : 0x0000;
00150             pc98_egc_fgcm[1].w = (val & 2) ? 0xFFFF : 0x0000;
00151             pc98_egc_fgcm[2].w = (val & 4) ? 0xFFFF : 0x0000;
00152             pc98_egc_fgcm[3].w = (val & 8) ? 0xFFFF : 0x0000;
00153             break;
00154         case 0x8: /* 0x4A8 */
00155             if (pc98_egc_compare_lead == 0)
00156                 *((uint16_t*)pc98_egc_mask) = val;
00157             break;
00158         case 0xA: /* 0x4AA */
00159             /* If FGC = 0 and BGC = 0:
00160              *   bits [15:0] = 0
00161              * If FGC = 1 or BGC = 1:
00162              *   bits [15:8] = 0
00163              *   bits [7:0] = foreground color (all 8 bits used in 256-color mode) */
00164             pc98_egc_background_color = val;
00165             pc98_egc_bgcm[0].w = (val & 1) ? 0xFFFF : 0x0000;
00166             pc98_egc_bgcm[1].w = (val & 2) ? 0xFFFF : 0x0000;
00167             pc98_egc_bgcm[2].w = (val & 4) ? 0xFFFF : 0x0000;
00168             pc98_egc_bgcm[3].w = (val & 8) ? 0xFFFF : 0x0000;
00169             break;
00170         case 0xC: /* 0x4AC */
00171             /* bits[15:13] = 0
00172              * bits[12:12] = shift direction 0=ascend 1=descend
00173              * bits[11:8] = 0
00174              * bits[7:4] = destination bit address
00175              * bits[3:0] = source bit address */
00176             pc98_egc_shift_descend = !!((val >> 12) & 1);
00177             pc98_egc_shift_destbit = (val >> 4) & 0xF;
00178             pc98_egc_shift_srcbit = val & 0xF;
00179             pc98_egc_shift_reinit();
00180             break;
00181         case 0xE: /* 0x4AE */
00182             /* bits[15:12] = 0
00183              * bits[11:0] = bit length (0 to 4095) */
00184             pc98_egc_shift_length = val & 0xFFF;
00185             pc98_egc_shift_reinit();
00186             break;
00187         default:
00188             // LOG_MSG("PC-98 EGC: Unhandled write to 0x%x val 0x%x",(unsigned int)port,(unsigned int)val);
00189             break;
00190     };
