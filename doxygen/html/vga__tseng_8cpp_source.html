00074 // Tseng ET4K implementation
00075 void write_p3d5_et4k(Bitu reg,Bitu val,Bitu iolen) {
00076         if(!et4k.extensionsEnabled && reg!=0x33)
00077                 return;
00078 
00079         switch(reg) {
00080         /*
00081         3d4h index 31h (R/W):  General Purpose
00082         bit  0-3  Scratch pad
00083              6-7  Clock Select bits 3-4. Bits 0-1 are in 3C2h/3CCh bits 2-3.
00084         */
00085                 STORE_ET4K(3d4, 31);
00086 
00087         // 3d4h index 32h - RAS/CAS Configuration (R/W)
00088         // No effect on emulation. Should not be written by software.
00089         STORE_ET4K(3d4, 32);
00090 
00091         case 0x33:
00092                 // 3d4 index 33h (R/W): Extended start Address
00093                 // 0-1 Display Start Address bits 16-17
00094                 // 2-3 Cursor start address bits 16-17
00095                 // Used by standard Tseng ID scheme
00096                 et4k.store_3d4_33 = val;
00097                 vga.config.display_start = (vga.config.display_start & 0xffff) | ((val & 0x03)<<16);
00098                 vga.config.cursor_start = (vga.config.cursor_start & 0xffff) | ((val & 0x0c)<<14);
00099                 break;
00100 
00101         /*
00102         3d4h index 34h (R/W): 6845 Compatibility Control Register
00103         bit    0  Enable CS0 (alternate clock timing)
00104                    1  Clock Select bit 2.  Bits 0-1 in 3C2h bits 2-3, bits 3-4 are in 3d4h
00105                           index 31h bits 6-7
00106                    2  Tristate ET4000 bus and color outputs if set
00107                    3  Video Subsystem Enable Register at 46E8h if set, at 3C3h if clear.
00108                    4  Enable Translation ROM for reading CRTC and MISCOUT if set
00109                    5  Enable Translation ROM for writing CRTC and MISCOUT if set
00110                    6  Enable double scan in AT&T compatibility mode if set
00111                    7  Enable 6845 compatibility if set
00112         */
00113         // TODO: Bit 6 may have effect on emulation
00114         STORE_ET4K(3d4, 34);
00115 
00116         case 0x35: 
00117         /*
00118         3d4h index 35h (R/W): Overflow High
00119         bit    0  Vertical Blank Start Bit 10 (3d4h index 15h).
00120                    1  Vertical Total Bit 10 (3d4h index 6).
00121                    2  Vertical Display End Bit 10 (3d4h index 12h).
00122                    3  Vertical Sync Start Bit 10 (3d4h index 10h).
00123                    4  Line Compare Bit 10 (3d4h index 18h).
00124                    5  Gen-Lock Enabled if set (External sync)
00125                    6  (4000) Read/Modify/Write Enabled if set. Currently not implemented.
00126                    7  Vertical interlace if set. The Vertical timing registers are
00127                         programmed as if the mode was non-interlaced!!
00128         */
00129                 et4k.store_3d4_35 = val;
00130                 vga.config.line_compare = (vga.config.line_compare & 0x3ff) | ((val&0x10)<<6);
00131         // Abusing s3 ex_ver_overflow field. This is to be cleaned up later.
00132                 {
00133                         Bit8u s3val =
00134                                 ((val & 0x01) << 2) | // vbstart
00135                                 ((val & 0x02) >> 1) | // vtotal
00136                                 ((val & 0x04) >> 1) | // vdispend
00137                                 ((val & 0x08) << 1) | // vsyncstart (?)
00138                                 ((val & 0x10) << 2); // linecomp
00139                         if ((s3val ^ vga.s3.ex_ver_overflow) & 0x3) {
00140                                 vga.s3.ex_ver_overflow=s3val;
00141                                 VGA_StartResize();
00142                         } else vga.s3.ex_ver_overflow=s3val;
00143                 }
00144                 break;
00145 
00146         // 3d4h index 36h - Video System Configuration 1 (R/W)
00147         // VGADOC provides a lot of info on this register, Ferraro has significantly less detail.
00148         // This is unlikely to be used by any games. Bit 4 switches chipset into linear mode -
00149         // that may be useful in some cases if there is any software actually using it.
00150         // TODO (not near future): support linear addressing
00151         STORE_ET4K(3d4, 36);
00152 
00153         // 3d4h index 37 - Video System Configuration 2 (R/W)
00154         // Bits 0,1, and 3 provides information about memory size:
00155         // 0-1 Bus width (1: 8 bit, 2: 16 bit, 3: 32 bit)
00156         // 3   Size of RAM chips (0: 64Kx, 1: 256Kx)
00157         // Other bits have no effect on emulation.
00158         case 0x37:
00159                 if (val != et4k.store_3d4_37) {
00160                         et4k.store_3d4_37 = val;
00161                         vga.vmemwrap = ((64*1024)<<((val&8)>>2))<<((val&3)-1);
00162                         VGA_SetupHandlers();
00163                 }
00164                 break;
00165 
00166         case 0x3f:
00167         /*
00168         3d4h index 3Fh (R/W):
00169         bit    0  Bit 8 of the Horizontal Total (3d4h index 0)
00170                    2  Bit 8 of the Horizontal Blank Start (3d4h index 3)
00171                    4  Bit 8 of the Horizontal Retrace Start (3d4h index 4)
00172                    7  Bit 8 of the CRTC offset register (3d4h index 13h).
00173         */
00174         // The only unimplemented one is bit 7
00175                 et4k.store_3d4_3f = val;
00176         // Abusing s3 ex_hor_overflow field which very similar. This is
00177         // to be cleaned up later
00178                 if ((val ^ vga.s3.ex_hor_overflow) & 3) {
00179                         vga.s3.ex_hor_overflow=(val&0x15);
00180                         VGA_StartResize();
00181                 } else vga.s3.ex_hor_overflow=(val&0x15);
00182                 break;
00183         default:
00184                 LOG(LOG_VGAMISC,LOG_NORMAL)("VGA:CRTC:ET4K:Write to illegal index %2X", (int)reg);
00185                 break;
00186         }
00187 }

00314 static Bitu get_clock_index_et4k() {
00315         // Ignoring bit 4, using "only" 16 frequencies. Looks like most implementations had only that
00316         return ((vga.misc_output>>2)&3) | ((et4k.store_3d4_34<<1)&4) | ((et4k.store_3d4_31>>3)&8);
00317 }

00319 static void set_clock_index_et4k(Bitu index) {
00320         // Shortwiring register reads/writes for simplicity
00321         IO_Write(0x3c2, (vga.misc_output&~0x0c)|((index&3)<<2));
00322         et4k.store_3d4_34 = (et4k.store_3d4_34&~0x02)|((index&4)>>1);
00323         et4k.store_3d4_31 = (et4k.store_3d4_31&~0xc0)|((index&8)<<3); // (index&0x18) if 32 clock frequencies are to be supported
00324 }

00326 void FinishSetMode_ET4K(Bitu crtc_base, VGA_ModeExtraData* modeData) {
00327         // Note that switching to a mode always puts the DAC in 15-bit color. An extra BIOS call is necessary to switch to 16-bit color.
00328         // We are also forcing the mode back to work exactly the way it did on the real hardware
00329         if (modeData->modeNo & 0x200) {
00330                 et4k.hicolorDACcommand = 0xa0;
00331                 modeData->modeNo &= ~0x200;
00332         } else {
00333                 et4k.hicolorDACcommand = 0x00;
00334         }
00335 
00336         et4k.biosMode = modeData->modeNo;
00337 
00338         IO_Write(0x3cd, 0x00); // both banks to 0
00339 
00340         // Reinterpret hor_overflow. Curiously, three bits out of four are
00341         // in the same places. Input has hdispend (not supported), output
00342         // has CRTC offset (also not supported)
00343         Bit8u et4k_hor_overflow = 
00344                 (modeData->hor_overflow & 0x01) |
00345                 (modeData->hor_overflow & 0x04) |
00346                 (modeData->hor_overflow & 0x10);
00347         IO_Write(crtc_base,0x3f);IO_Write(crtc_base+1,et4k_hor_overflow);
00348 
00349         // Reinterpret ver_overflow
00350         Bit8u et4k_ver_overflow =
00351                 ((modeData->ver_overflow & 0x01) << 1) | // vtotal10
00352                 ((modeData->ver_overflow & 0x02) << 1) | // vdispend10
00353                 ((modeData->ver_overflow & 0x04) >> 2) | // vbstart10
00354                 ((modeData->ver_overflow & 0x10) >> 1) | // vretrace10 (tseng has vsync start?)
00355                 ((modeData->ver_overflow & 0x40) >> 2);  // line_compare
00356         IO_Write(crtc_base,0x35);IO_Write(crtc_base+1,et4k_ver_overflow);
00357 
00358         // Clear remaining ext CRTC registers
00359         IO_Write(crtc_base,0x31);IO_Write(crtc_base+1,0);
00360         IO_Write(crtc_base,0x32);IO_Write(crtc_base+1,0);
00361         IO_Write(crtc_base,0x33);IO_Write(crtc_base+1,0);
00362         IO_Write(crtc_base,0x34);IO_Write(crtc_base+1,0);
00363         IO_Write(crtc_base,0x36);IO_Write(crtc_base+1,0);
00364         IO_Write(crtc_base,0x37);IO_Write(crtc_base+1,0x0c|(vga.vmemsize==1024*1024?3:vga.vmemsize==512*1024?2:1));
00365         // Clear ext SEQ
00366         IO_Write(0x3c4,0x06);IO_Write(0x3c5,0);
00367         IO_Write(0x3c4,0x07);IO_Write(0x3c5,0);
00368         // Clear ext ATTR
00369         IO_Write(0x3c0,0x16);IO_Write(0x3c0,0);
00370         IO_Write(0x3c0,0x17);IO_Write(0x3c0,0);
00371 
00372         // Select SVGA clock to get close to 60Hz (not particularly clean implementation)
00373         if (modeData->modeNo > 0x13) {
00374                 Bitu target = modeData->vtotal*8*modeData->htotal*60;
00375                 Bitu best = 1;
00376                 Bits dist = 100000000;
00377                 for (Bitu i=0; i<16; i++) {
00378                         Bits cdiff=abs((Bits)(target-et4k.clockFreq[i]));
00379                         if (cdiff < dist) {
00380                                 best = i;
00381                                 dist = cdiff;
00382                         }
00383                 }
00384                 set_clock_index_et4k(best);
00385         }
00386 
00387         if(svga.determine_mode)
00388                 svga.determine_mode();
00389 
00390         // Verified (on real hardware and in a few games): Tseng ET4000 used chain4 implementation
00391         // different from standard VGA. It was also not limited to 64K in regular mode 13h.
00392         vga.config.compatible_chain4 = false;
00393         vga.vmemwrap = vga.vmemsize;
00394 
00395         VGA_SetupHandlers();
00396 }

00398 void DetermineMode_ET4K() {
00399         // Special case for HiColor DAC enabled modes
00400         if ((et4k.hicolorDACcommand & 0xc0) == 0x80) {
00401                 VGA_SetMode(vga_sierra_lock_565 ? M_LIN16 : M_LIN15);
00402                 return;
00403         } else if ((et4k.hicolorDACcommand & 0xc0) == 0xc0) {
00404                 VGA_SetMode(M_LIN16);
00405                 return;
00406         }
00407         // Close replica from the base implementation. It will stay here
00408         // until I figure a way to either distinguish M_VGA and M_LIN8 or
00409         // merge them.
00410         if (vga.attr.mode_control & 1) {
00411                 if (vga.gfx.mode & 0x40) VGA_SetMode((et4k.biosMode<=0x13)?M_VGA:M_LIN8); // Ugly...
00412                 else if (vga.gfx.mode & 0x20) VGA_SetMode(M_CGA4);
00413                 else if ((vga.gfx.miscellaneous & 0x0c)==0x0c) VGA_SetMode(M_CGA2);
00414                 else VGA_SetMode((et4k.biosMode<=0x13)?M_EGA:M_LIN4);
00415         } else {
00416                 VGA_SetMode(M_TEXT);
00417         }
00418 }

00434 // Support for HiColor DAC
00435 // Support is sufficient for proper detection (WHATVGA detects UMC 70c178, similar to Sierra SC11487, slightly simpler to work with).
00436 // Control over 15/16-bit color is also implemented. Need better documentation and/or test cases to advance the implementation.
00437 /*
00438 REG06 (R/W):  Command Register
00439 bit   0  (SC11487) (R) Set if bits 5-7 is 1 or 3, clear otherwise
00440     3-4  (SC11487) Accesses bits 3-4 of the PEL Mask registers (REG02)
00441       5  (not SC11481/6/8)
00442          If set two pixel clocks are used to latch the two bytes
00443          needed for each pixel. Low byte is latched first.
00444          If clear the low byte is latched on the rising edge of the
00445          pixel clock and the high byte is latched on the falling edge.
00446          Only some VGA chips (ET4000 and C&T655x0) can handle this.
00447       6  (SC11485/7/9, OTI66HC, UM70C178) Set in 16bit (64k) modes (Only valid
00448            if bit 7 set). On the SC11482/3/4 this bit is read/writable, but
00449            has no function. On the SC11481/6/8 this bit does not exist.
00450       7  Set in HiColor (32k/64k) modes, clear in palette modes.
00451 Note:  This register can also be accessed at 3C6h by reading 3C6h four times,
00452        then all accesses to 3C6h will go the this register until one of the
00453        registers 3C7h, 3C8h or 3C9h is accessed.
00454 */
