01120 static bool IsEnhancedKey(Bit16u &key) {
01121     if (IS_PC98_ARCH)
01122         return false;
01123 
01124         /* test for special keys (return and slash on numblock) */
01125         if ((key>>8)==0xe0) {
01126                 if (((key&0xff)==0x0a) || ((key&0xff)==0x0d)) {
01127                         /* key is return on the numblock */
01128                         key=(key&0xff)|0x1c00;
01129                 } else {
01130                         /* key is slash on the numblock */
01131                         key=(key&0xff)|0x3500;
01132                 }
01133                 /* both keys are not considered enhanced keys */
01134                 return false;
01135         } else if (((key>>8)>0x84) || (((key&0xff)==0xf0) && (key>>8))) {
01136                 /* key is enhanced key (either scancode part>0x84 or
01137                    specially-marked keyboard combination, low part==0xf0) */
01138                 return true;
01139         }
01140         /* convert key if necessary (extended keys) */
01141         if ((key>>8) && ((key&0xff)==0xe0))  {
01142                 key&=0xff00;
01143         }
01144         return false;

01161                         /* enter small idle loop to allow for irqs to happen */
01162                         reg_ip+=1;
01163                 }
01164                 break;
01165         case 0x10: /* GET KEYSTROKE (enhanced keyboards only) */
01166         if (int16_unmask_irq1_on_read)
01167             PIC_SetIRQMask(1,false); /* unmask keyboard */
01168 
01169                 if (get_key(temp)) {
01170                         if (!IS_PC98_ARCH && ((temp&0xff)==0xf0) && (temp>>8)) {
01171                                 /* special enhanced key, clear low part before returning key */
01172                                 temp&=0xff00;
01173                         }
01174                         reg_ax=temp;
01175                 } else {
01176                         /* enter small idle loop to allow for irqs to happen */
01177                         reg_ip+=1;
01178                 }
01179                 break;
01180         case 0x01: /* CHECK FOR KEYSTROKE */
01181                 // enable interrupt-flag after IRET of this int16
01182                 CALLBACK_SIF(true);
01183         if (int16_unmask_irq1_on_read)
01184             PIC_SetIRQMask(1,false); /* unmask keyboard */
01185 
01186                 for (;;) {
01187                         if (check_key(temp)) {
01188                                 if (!IsEnhancedKey(temp)) {
01189                                         /* normal key, return translated key in ax */
01190                                         CALLBACK_SZF(false);
01191                     if (int16_ah_01_cf_undoc) CALLBACK_SCF(true);
01192                                         reg_ax=temp;
01193                                         break;
01194                                 } else {
01195                                         /* remove enhanced key from buffer and ignore it */
01196                                         get_key(temp);
01197                                 }
01198                         } else {
01199                                 /* no key available */
01200                                 CALLBACK_SZF(true);
01201                 if (int16_ah_01_cf_undoc) CALLBACK_SCF(false);
01202                                 break;
01203                         }
01204 //                      CALLBACK_Idle();
01205                 }
01206                 break;
01207         case 0x11: /* CHECK FOR KEYSTROKE (enhanced keyboards only) */
01208                 // enable interrupt-flag after IRET of this int16
01209                 CALLBACK_SIF(true);
01210         if (int16_unmask_irq1_on_read)
01211             PIC_SetIRQMask(1,false); /* unmask keyboard */
01212 
01213                 if (!check_key(temp)) {
01214                         CALLBACK_SZF(true);
01215                 } else {
01216                         CALLBACK_SZF(false);
01217                         if (!IS_PC98_ARCH && ((temp&0xff)==0xf0) && (temp>>8)) {
01218                                 /* special enhanced key, clear low part before returning key */
01219                                 temp&=0xff00;
01220                         }
01221                         reg_ax=temp;
01222                 }
01223                 break;
01224         case 0x02:      /* GET SHIFT FLAGS */
01225                 reg_al=mem_readb(BIOS_KEYBOARD_FLAGS1);
01226                 break;
01227         case 0x03:      /* SET TYPEMATIC RATE AND DELAY */
01228                 if (reg_al == 0x00) { // set default delay and rate
01229                         IO_Write(0x60,0xf3);
01230                         IO_Write(0x60,0x20); // 500 msec delay, 30 cps
01231                 } else if (reg_al == 0x05) { // set repeat rate and delay
01232                         IO_Write(0x60,0xf3);

01239                 if (BIOS_AddKeyToBuffer(reg_cx)) reg_al=0;
01240                 else reg_al=1;
01241                 break;
01242         case 0x12: /* GET EXTENDED SHIFT STATES */

01246                          (mem_readb(BIOS_KEYBOARD_FLAGS3)&0x0c);    // Right Ctrl/Alt pressed, bits 2,3

01249                 /* Weird call used by some dos apps */

01251                 break;
