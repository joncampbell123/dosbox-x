00067         void PlayEvent(Bit8u *msg, Bitu len) {
00068                 Bit8u event = msg[0], channel, p1, p2;
00069 
00070                 switch (event) {
00071                 case 0xf0:
00072                 case 0xf7:
00073                         LOG(LOG_MISC,LOG_DEBUG)("SYNTH: sysex 0x%02x len %lu", (int)event, (long unsigned)len);
00074                         fluid_synth_sysex(synth_soft, (char *)(msg + 1), len - 1, NULL, NULL, NULL, 0);
00075                         return;
00076                 case 0xf9:
00077                         LOG(LOG_MISC,LOG_DEBUG)("SYNTH: midi tick");
00078                         return;
00079                 case 0xff:
00080                         LOG(LOG_MISC,LOG_DEBUG)("SYNTH: system reset");
00081                         fluid_synth_system_reset(synth_soft);
00082                         return;
00083                 case 0xf1: case 0xf2: case 0xf3: case 0xf4:
00084                 case 0xf5: case 0xf6: case 0xf8: case 0xfa:
00085                 case 0xfb: case 0xfc: case 0xfd: case 0xfe:
00086                         LOG(LOG_MISC,LOG_WARN)("SYNTH: unhandled event 0x%02x", (int)event);
00087                         return;
00088                 }
00089 
00090                 channel = event & 0xf;
00091                 p1 = len > 1 ? msg[1] : 0;
00092                 p2 = len > 2 ? msg[2] : 0;
00093 
00094                 LOG(LOG_MISC,LOG_DEBUG)("SYNTH: event 0x%02x channel %d, 0x%02x 0x%02x",
00095                         (int)event, (int)channel, (int)p1, (int)p2);
00096 
00097                 switch (event & 0xf0) {
00098                 case 0x80:
00099                         fluid_synth_noteoff(synth_soft, channel, p1);
00100                         break;
00101                 case 0x90:
00102                         fluid_synth_noteon(synth_soft, channel, p1, p2);
00103                         break;
00104                 case 0xb0:
00105                         fluid_synth_cc(synth_soft, channel, p1, p2);
00106                         break;
00107                 case 0xc0:
00108                         fluid_synth_program_change(synth_soft, channel, p1);
00109                         break;
00110                 case 0xd0:
00111                         fluid_synth_channel_pressure(synth_soft, channel, p1);
00112                         break;
00113                 case 0xe0:
00114                         fluid_synth_pitch_bend(synth_soft, channel, (p2 << 7) | p1);
00115                         break;
00116                 }
00117         };
