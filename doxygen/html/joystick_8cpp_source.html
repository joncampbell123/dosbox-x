00054 static Bitu read_p201(Bitu port,Bitu iolen) {
00055         /* Reset Joystick to 0 after TIMEOUT ms */
00056         if(write_active && ((PIC_Ticks - last_write) > TIMEOUT)) {
00057                 write_active = false;
00058                 stick[0].xcount = 0;
00059                 stick[1].xcount = 0;
00060                 stick[0].ycount = 0;
00061                 stick[1].ycount = 0;
00062 //              LOG_MSG("reset by time %d %d",PIC_Ticks,last_write);
00063         }
00064 
00074         Bit8u ret=0xff;
00075         if (stick[0].enabled) {
00076                 if (stick[0].xcount) stick[0].xcount--; else ret&=~1;
00077                 if (stick[0].ycount) stick[0].ycount--; else ret&=~2;
00078                 if (stick[0].button[0]) ret&=~16;
00079                 if (stick[0].button[1]) ret&=~32;
00080         }
00081         if (stick[1].enabled) {
00082                 if (stick[1].xcount) stick[1].xcount--; else ret&=~4;
00083                 if (stick[1].ycount) stick[1].ycount--; else ret&=~8;
00084                 if (stick[1].button[0]) ret&=~64;
00085                 if (stick[1].button[1]) ret&=~128;
00086         }
00087         return ret;
00088 }

00127 static void write_p201_timed(Bitu port,Bitu val,Bitu iolen) {
00128         // Store writetime index
00129         // Axes take time = 24.2 microseconds + ( 0.011 microsecons/ohm * resistance )
00130         // to reset to 0
00131         // Precalculate the time at which each axis hits 0 here
00132         double currentTick = PIC_FullIndex();
00133         if (stick[0].enabled) {
00134                 stick[0].xtick = currentTick + 1000.0*( JOY_S_CONSTANT + S_PER_OHM *
00135                                  (double)(((stick[0].xpos+1.0)* OHMS)) );
00136                 stick[0].ytick = currentTick + 1000.0*( JOY_S_CONSTANT + S_PER_OHM *
00137                                  (double)(((stick[0].ypos+1.0)* OHMS)) );
00138         }
00139         if (stick[1].enabled) {
00140                 stick[1].xtick = currentTick + 1000.0*( JOY_S_CONSTANT + S_PER_OHM *
00141                                  (double)((swap34? stick[1].ypos : stick[1].xpos)+1.0) * OHMS);
00142                 stick[1].ytick = currentTick + 1000.0*( JOY_S_CONSTANT + S_PER_OHM *
00143                                  (double)((swap34? stick[1].xpos : stick[1].ypos)+1.0) * OHMS);
00144         }
