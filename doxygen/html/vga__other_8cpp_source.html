00038 static void write_crtc_data_other(Bitu /*port*/,Bitu val,Bitu /*iolen*/) {
00039         switch (vga.other.index) {
00040         case 0x00:              //Horizontal total
00041                 if (vga.other.htotal ^ val) VGA_StartResize();
00042                 vga.other.htotal=(Bit8u)val;
00043                 break;
00044         case 0x01:              //Horizontal displayed chars
00045                 if (vga.other.hdend ^ val) VGA_StartResize();
00046                 vga.other.hdend=(Bit8u)val;
00047                 break;
00048         case 0x02:              //Horizontal sync position
00049                 vga.other.hsyncp=(Bit8u)val;
00050                 break;
00051         case 0x03:              //Horizontal sync width
00052                 if (machine==MCH_TANDY) vga.other.vsyncw=(Bit8u)(val >> 4);
00053                 else vga.other.vsyncw = 16; // The MC6845 has a fixed v-sync width of 16 lines
00054                 vga.other.hsyncw=(Bit8u)(val & 0xf);
00055                 break;
00056         case 0x04:              //Vertical total
00057                 if (vga.other.vtotal ^ val) VGA_StartResize();
00058                 vga.other.vtotal=(Bit8u)(val&0x7f);
00059                 break;
00060         case 0x05:              //Vertical display adjust
00061                 if (vga.other.vadjust ^ val) VGA_StartResize();
00062                 vga.other.vadjust=(Bit8u)val;
00063                 break;
00064         case 0x06:              //Vertical rows
00065                 if (vga.other.vdend ^ val) VGA_StartResize();
00066                 vga.other.vdend=(Bit8u)(val&0x7f);
00067                 break;
00068         case 0x07:              //Vertical sync position
00069                 vga.other.vsyncp=(Bit8u)val;
00070                 break;
00071         case 0x09:              //Max scanline
00072                 val &= 0x1f; // VGADOC says bit 0-3 but the MC6845 datasheet says bit 0-4
00073                 if (vga.other.max_scanline ^ val) VGA_StartResize();
00074                 vga.other.max_scanline=(Bit8u)val;
00075                 break;
00076         case 0x0A:      /* Cursor Start Register */
00077                 vga.other.cursor_start = (Bit8u)(val & 0x3f);
00078                 vga.draw.cursor.sline = (Bit8u)(val&0x1f);
00079                 vga.draw.cursor.enabled = ((val & 0x60) != 0x20);
00080                 break;
00081         case 0x0B:      /* Cursor End Register */
00082                 vga.other.cursor_end = (Bit8u)(val&0x1f);
00083                 vga.draw.cursor.eline = (Bit8u)(val&0x1f);
00084                 break;
00085         case 0x0C:      /* Start Address High Register */
00086                 // Bit 12 (depending on video mode) and 13 are actually masked too,
00087                 // but so far no need to implement it.
00088                 vga.config.display_start=(vga.config.display_start & 0x00FF) | ((val&0x3F) << 8);
00089                 break;
00090         case 0x0D:      /* Start Address Low Register */
00091                 vga.config.display_start=(vga.config.display_start & 0xFF00) | val;
00092                 break;
00093         case 0x0E:      /*Cursor Location High Register */
00094                 vga.config.cursor_start&=0x00ff;
00095                 vga.config.cursor_start|=((Bit8u)val) << 8;
00096                 break;
00097         case 0x0F:      /* Cursor Location Low Register */
00098                 vga.config.cursor_start&=0xff00;
00099                 vga.config.cursor_start|=(Bit8u)val;
00100                 break;
00101         case 0x10:      /* Light Pen High */
00102                 // MC6845 datasheet says the light pen registers are only readable
00103                 vga.other.lightpen &= 0xff;
00104                 vga.other.lightpen |= (val & 0x3f)<<8;          // only 6 bits
00105                 break;
00106         case 0x11:      /* Light Pen Low */
00107                 vga.other.lightpen &= 0xff00;
00108                 vga.other.lightpen |= (Bit8u)val;
00109                 break;
00110         default:
00111                 LOG(LOG_VGAMISC,LOG_NORMAL)("MC6845:Write %X to illegal index %x",(int)val,(int)vga.other.index);
00112         }

00230 static void update_cga16_color(void) {
00231 // New algorithm based on code by reenigne
00232 // Works in all CGA graphics modes/color settings and can simulate older and newer CGA revisions
00233         static const double tau = 6.28318531; // == 2*pi
00234         static const double ns = 567.0/440;  // degrees of hue shift per nanosecond
00235 
00236         double tv_brightness = 0.0; // hardcoded for simpler implementation
00237         double tv_saturation = (new_cga ? 0.7 : 0.6);
00238 
00239         bool bw = (vga.tandy.mode_control&4) != 0;
00240         bool color_sel = (cga16_val&0x20) != 0;
00241         bool background_i = (cga16_val&0x10) != 0;      // Really foreground intensity, but this is what the CGA schematic calls it.
00242         bool bpp1 = (vga.tandy.mode_control&0x10) != 0;
00243         Bit8u overscan = cga16_val&0x0f;  // aka foreground colour in 1bpp mode
00244 
00245         double chroma_coefficient = new_cga ? 0.29 : 0.72;
00246         double b_coefficient = new_cga ? 0.07 : 0;
00247         double g_coefficient = new_cga ? 0.22 : 0;
00248         double r_coefficient = new_cga ? 0.1 : 0;
00249         double i_coefficient = new_cga ? 0.32 : 0.28;
00250         double rgbi_coefficients[0x10];
00251         for (int c = 0; c < 0x10; c++) {
00252                 double v = 0;
00253                 if ((c & 1) != 0)
00254                         v += b_coefficient;
00255                 if ((c & 2) != 0)
00256                         v += g_coefficient;
00257                 if ((c & 4) != 0)
00258                         v += r_coefficient;
00259                 if ((c & 8) != 0)
00260                         v += i_coefficient;
00261                 rgbi_coefficients[c] = v;
00262         }
00263 
00264         // The pixel clock delay calculation is not accurate for 2bpp, but the difference is small and a more accurate calculation would be too slow.
00265         static const double rgbi_pixel_delay = 15.5*ns;
00266         static const double chroma_pixel_delays[8] = {
00267                 0,        // Black:   no chroma
00268                 35*ns,    // Blue:    no XORs
00269                 44.5*ns,  // Green:   XOR on rising and falling edges
00270                 39.5*ns,  // Cyan:    XOR on falling but not rising edge
00271                 44.5*ns,  // Red:     XOR on rising and falling edges
00272                 39.5*ns,  // Magenta: XOR on falling but not rising edge
00273                 44.5*ns,  // Yellow:  XOR on rising and falling edges
00274                 39.5*ns}; // White:   XOR on falling but not rising edge
00275         double pixel_clock_delay;
00276         int o = overscan == 0 ? 15 : overscan;
00277         if (overscan == 8)
00278                 pixel_clock_delay = rgbi_pixel_delay;
00279         else {
00280                 double d = rgbi_coefficients[o];
00281                 pixel_clock_delay = (chroma_pixel_delays[o & 7]*chroma_coefficient + rgbi_pixel_delay*d)/(chroma_coefficient + d);
00282         }
00283         pixel_clock_delay -= 21.5*ns;  // correct for delay of color burst
00284 
00285         double hue_adjust = (-(90-33)-hue_offset+pixel_clock_delay)*tau/360.0;
00286         double chroma_signals[8][4];
00287         for (Bit8u i=0; i<4; i++) {
00288                 chroma_signals[0][i] = 0;
00289                 chroma_signals[7][i] = 1;
00290                 for (Bit8u j=0; j<6; j++) {
00291                         static const double phases[6] = {
00292                                 270 - 21.5*ns,  // blue
00293                                 135 - 29.5*ns,  // green
00294                                 180 - 21.5*ns,  // cyan
00295                                   0 - 21.5*ns,  // red
00296                                 315 - 29.5*ns,  // magenta
00297                                  90 - 21.5*ns}; // yellow/burst
00298                         // All the duty cycle fractions are the same, just under 0.5 as the rising edge is delayed 2ns more than the falling edge.
00299                         static const double duty = 0.5 - 2*ns/360.0;
00300 
00301                         // We have a rectangle wave with period 1 (in units of the reciprocal of the color burst frequency) and duty
00302                         // cycle fraction "duty" and phase "phase". We band-limit this wave to frequency 2 and sample it at intervals of 1/4.
00303                         // We model our band-limited wave with 4 frequency components:
00304                         //   f(x) = a + b*sin(x*tau) + c*cos(x*tau) + d*sin(x*2*tau)
00305                         // Then:
00306                         //   a =   integral(0, 1, f(x)*dx) = duty
00307                         //   b = 2*integral(0, 1, f(x)*sin(x*tau)*dx) = 2*integral(0, duty, sin(x*tau)*dx) = 2*(1-cos(x*tau))/tau
00308                         //   c = 2*integral(0, 1, f(x)*cos(x*tau)*dx) = 2*integral(0, duty, cos(x*tau)*dx) = 2*sin(duty*tau)/tau
00309                         //   d = 2*integral(0, 1, f(x)*sin(x*2*tau)*dx) = 2*integral(0, duty, sin(x*4*pi)*dx) = 2*(1-cos(2*tau*duty))/(2*tau)
00310                         double a = duty;
00311                         double b = 2.0*(1.0-cos(duty*tau))/tau;
00312                         double c = 2.0*sin(duty*tau)/tau;
00313                         double d = 2.0*(1.0-cos(duty*2*tau))/(2*tau);
00314 
00315                         double x = (phases[j] + 21.5*ns + pixel_clock_delay)/360.0 + i/4.0;
00316 
00317                         chroma_signals[j+1][i] = a + b*sin(x*tau) + c*cos(x*tau) + d*sin(x*2*tau);
00318                 }
00319         }
00320         Bitu CGApal[4] = {
00321                 overscan,
00322                 (Bitu)(2 + (color_sel||bw ? 1 : 0) + (background_i ? 8 : 0)),
00323                 (Bitu)(4 + (color_sel&&!bw? 1 : 0) + (background_i ? 8 : 0)),
00324                 (Bitu)(6 + (color_sel||bw ? 1 : 0) + (background_i ? 8 : 0))
00325         };
00326         for (Bit8u x=0; x<4; x++) {      // Position of pixel in question
00327                 bool even = (x & 1) == 0;
00328                 for (Bit8u bits=0; bits<(even ? 0x10 : 0x40); ++bits) {
00329                         double Y=0, I=0, Q=0;
00330                         for (Bit8u p=0; p<4; p++) {  // Position within color carrier cycle
00331                                 // generate pixel pattern.
00332                                 Bit8u rgbi;
00333                                 if (bpp1)
00334                                         rgbi = ((bits >> (3-p)) & (even ? 1 : 2)) != 0 ? overscan : 0;
00335                                 else
00336                                         if (even)
00337                                                 rgbi = CGApal[(bits >> (2-(p&2)))&3];
00338                                         else
00339                                                 rgbi = CGApal[(bits >> (4-((p+1)&6)))&3];
00340                                 Bit8u c = rgbi & 7;
00341                                 if (bw && c != 0)
00342                                         c = 7;
00343 
00344                                 // calculate composite output
00345                                 double chroma = chroma_signals[c][(p+x)&3]*chroma_coefficient;
00346                                 double composite = chroma + rgbi_coefficients[rgbi];
00347 
00348                                 Y+=composite;
00349                                 if (!bw) { // burst on
00350                                         I+=composite*2*cos(hue_adjust + (p+x)*tau/4.0);
00351                                         Q+=composite*2*sin(hue_adjust + (p+x)*tau/4.0);
00352                                 }
00353                         }
00354 
00355                         double contrast = 1 - tv_brightness;
00356 
00357                         Y = (contrast*Y/4.0) + tv_brightness; if (Y>1.0) Y=1.0; if (Y<0.0) Y=0.0;
00358                         I = (contrast*I/4.0) * tv_saturation; if (I>0.5957) I=0.5957; if (I<-0.5957) I=-0.5957;
00359                         Q = (contrast*Q/4.0) * tv_saturation; if (Q>0.5226) Q=0.5226; if (Q<-0.5226) Q=-0.5226;
00360 
00361                         static const double gamma = 2.2;
00362 
00363                         double R = Y + 0.9563*I + 0.6210*Q;     R = (R - 0.075) / (1-0.075); if (R<0) R=0; if (R>1) R=1;
00364                         double G = Y - 0.2721*I - 0.6474*Q;     G = (G - 0.075) / (1-0.075); if (G<0) G=0; if (G>1) G=1;
00365                         double B = Y - 1.1069*I + 1.7046*Q;     B = (B - 0.075) / (1-0.075); if (B<0) B=0; if (B>1) B=1;
00366                         R = pow(R, gamma);
00367                         G = pow(G, gamma);
00368                         B = pow(B, gamma);

00676                 // Bit 3-5: Processor page CPU_PG

00691 static void write_pcjr(Bitu port,Bitu val,Bitu /*iolen*/) {
00692         switch (port) {
00693         case 0x3da:
00694                 if (vga.tandy.pcjr_flipflop) write_tandy_reg((Bit8u)val);
00695                 else {
00696                         vga.tandy.reg_index=(Bit8u)val;
00697                         if (vga.tandy.reg_index & 0x10)
00698                                 vga.attr.disabled |= 2;
00699                         else vga.attr.disabled &= ~2;
00700                 }
00701                 vga.tandy.pcjr_flipflop=!vga.tandy.pcjr_flipflop;
00702                 break;
00703         case 0x3df:
00704                 // CRT/processor page register
00705                 
00706                 // Bit 0-2: CRT page PG0-2
00707                 // In one- and two bank modes, bit 0-2 select the 16kB memory
00708                 // area of system RAM that is displayed on the screen.
00709                 // In 4-banked modes, bit 1-2 select the 32kB memory area.
00710                 // Bit 2 only has effect when the PCJR upgrade to 128k is installed.
00711                 
00712                 // Bit 3-5: Processor page CPU_PG
00713                 // Selects the 16kB area of system RAM that is mapped to
00714                 // the B8000h IBM PC video memory window. Since A14-A16 of the 
00715                 // processor are unconditionally replaced with these bits when
00716                 // B8000h is accessed, the 16kB area is mapped to the 32kB
00717                 // range twice in a row. (Scuba Venture writes across the boundary)
00718                 
00719                 // Bit 6-7: Video Address mode
00720                 // 0: CRTC addresses A0-12 directly, accessing 8k characters
00721                 //    (+8k attributes). Used in text modes (one bank).
00722                 //    PG0-2 in effect. 16k range.
00723                 // 1: CRTC A12 is replaced with CRTC RA0 (see max_scanline).
00724                 //    This results in the even/odd scanline two bank system.
00725                 //    PG0-2 in effect. 16k range.
00726                 // 2: Documented as unused. CRTC addresses A0-12, PG0 is replaced
00727                 //    with RA1. Looks like nonsense.
00728                 //    PG1-2 in effect. 32k range which cannot be used completely.
00729                 // 3: CRTC A12 is replaced with CRTC RA0, PG0 is replaced with
00730                 //    CRTC RA1. This results in the 4-bank mode.
00731                 //    PG1-2 in effect. 32k range.
00732 
00733                 vga.tandy.line_mask = (Bit8u)(val >> 6);
00734                 vga.tandy.draw_bank = val & ((vga.tandy.line_mask&2) ? 0x6 : 0x7);
00735                 vga.tandy.mem_bank = (val >> 3) & 7;
00736                 vga.tandy.draw_base = &MemBase[vga.tandy.draw_bank * 16 * 1024];
00737                 vga.tandy.mem_base = &MemBase[vga.tandy.mem_bank * 16 * 1024];
00738                 TandyCheckLineMask();
00739                 VGA_SetupHandlers();
00740                 break;
00741         }
00742 }

00852 Bitu read_herc_status(Bitu /*port*/,Bitu /*iolen*/) {
00853         // 3BAh (R):  Status Register
00854         // bit   0  Horizontal sync
00855         //       1  Light pen status (only some cards)
00856         //       3  Video signal
00857         //     4-6      000: Hercules
00858         //                      001: Hercules Plus
00859         //                      101: Hercules InColor
00860         //                      111: Unknown clone
00861         //       7  Vertical sync inverted
00862 
00863         double timeInFrame = PIC_FullIndex()-vga.draw.delay.framestart;
00864         Bit8u retval=0x72; // Hercules ident; from a working card (Winbond W86855AF)
00865                                         // Another known working card has 0x76 ("KeysoGood", full-length)
00866         if (timeInFrame < vga.draw.delay.vrstart ||
00867                 timeInFrame > vga.draw.delay.vrend) retval |= 0x80;
00868 
00869         double timeInLine=fmod(timeInFrame,vga.draw.delay.htotal);
00870         if (timeInLine >= vga.draw.delay.hrstart &&
00871                 timeInLine <= vga.draw.delay.hrend) retval |= 0x1;
00872 
00873         // 688 Attack sub checks bit 3 - as a workaround have the bit enabled
00874         // if no sync active (corresponds to a completely white screen)
00875         if ((retval&0x81)==0x80) retval |= 0x8;
00876         return retval;
00877 }
