00089 void SN76496Write(struct SN76496 *R,Bitu port,Bitu data) {
00090         /* update the output buffer before changing the registers */
00091 
00092         if (data & 0x80)
00093         {
00094                 int r = (data & 0x70) >> 4;
00095                 int c = r/2;
00096 
00097                 R->LastRegister = r;
00098                 R->Register[r] = (R->Register[r] & 0x3f0) | (data & 0x0f);
00099                 switch (r)
00100                 {
00101                         case 0: /* tone 0 : frequency */
00102                         case 2: /* tone 1 : frequency */
00103                         case 4: /* tone 2 : frequency */
00104                                 R->Period[c] = R->UpdateStep * R->Register[r];
00105                                 if (R->Period[c] == 0) R->Period[c] = 0x3fe;
00106                                 if (r == 4)
00107                                 {
00108                                         /* update noise shift frequency */
00109                                         if ((R->Register[6] & 0x03) == 0x03)
00110                                                 R->Period[3] = 2 * R->Period[2];
00111                                 }
00112                                 break;
00113                         case 1: /* tone 0 : volume */
00114                         case 3: /* tone 1 : volume */
00115                         case 5: /* tone 2 : volume */
00116                         case 7: /* noise  : volume */
00117                                 R->Volume[c] = R->VolTable[data & 0x0f];
00118                                 break;
00119                         case 6: /* noise  : frequency, mode */
00120                                 {
00121                                         int n = R->Register[6];
00122                                         R->NoiseFB = (n & 4) ? FB_WNOISE : FB_PNOISE;
00123                                         n &= 3;
00124                                         /* N/512,N/1024,N/2048,Tone #3 output */
00125                                         R->Period[3] = (n == 3) ? 2 * R->Period[2] : (R->UpdateStep << (5+n));
00126 
00127                                         /* reset noise shifter */
00128 //                                      R->RNG = NG_PRESET;
00129 //                                      R->Output[3] = R->RNG & 1;
00130                                 }
00131                                 break;
00132                 }
00133         }
00134         else
00135         {
00136                 int r = R->LastRegister;
00137                 int c = r/2;
00138 
00139                 switch (r)
00140                 {
00141                         case 0: /* tone 0 : frequency */
00142                         case 2: /* tone 1 : frequency */
00143                         case 4: /* tone 2 : frequency */
00144                                 R->Register[r] = (R->Register[r] & 0x0f) | ((data & 0x3f) << 4);
00145                                 R->Period[c] = R->UpdateStep * R->Register[r];
00146                                 if (R->Period[c] == 0) R->Period[c] = 0x3fe;
00147                                 if (r == 4)
00148                                 {
00149                                         /* update noise shift frequency */
00150                                         if ((R->Register[6] & 0x03) == 0x03)
00151                                                 R->Period[3] = 2 * R->Period[2];
00152                                 }
00153                                 break;
00154                 }
00155         }
00156 }

00158 void SN76496Update(struct SN76496 *R, Bit16s *buffer, Bitu length) {
00159         int i;
00160 
00161         /* If the volume is 0, increase the counter */
00162         for (i = 0;i < 4;i++)
00163         {
00164                 if (R->Volume[i] == 0)
00165                 {
00166                         /* note that I do count += length, NOT count = length + 1. You might think */
00167                         /* it's the same since the volume is 0, but doing the latter could cause */
00168                         /* interferencies when the program is rapidly modulating the volume. */
00169                         if (R->Count[i] <= (int)length*STEP) R->Count[i] += length*STEP;
00170                 }
00171         }
00172 
00173         Bitu count=length;
00174         while (count)
00175         {
00176                 int vol[4];
00177                 unsigned int out;
00178                 int left;
00179 
00180 
00181                 /* vol[] keeps track of how long each square wave stays */
00182                 /* in the 1 position during the sample period. */
00183                 vol[0] = vol[1] = vol[2] = vol[3] = 0;
00184 
00185                 for (i = 0;i < 3;i++)
00186                 {
00187                         if (R->Output[i]) vol[i] += R->Count[i];
00188                         R->Count[i] -= STEP;
00189                         /* Period[i] is the half period of the square wave. Here, in each */
00190                         /* loop I add Period[i] twice, so that at the end of the loop the */
00191                         /* square wave is in the same status (0 or 1) it was at the start. */
00192                         /* vol[i] is also incremented by Period[i], since the wave has been 1 */
00193                         /* exactly half of the time, regardless of the initial position. */
00194                         /* If we exit the loop in the middle, Output[i] has to be inverted */
00195                         /* and vol[i] incremented only if the exit status of the square */
00196                         /* wave is 1. */
00197                         while (R->Count[i] <= 0)
00198                         {
00199                                 R->Count[i] += R->Period[i];
00200                                 if (R->Count[i] > 0)
00201                                 {
00202                                         R->Output[i] ^= 1;
00203                                         if (R->Output[i]) vol[i] += R->Period[i];
00204                                         break;
00205                                 }
00206                                 R->Count[i] += R->Period[i];
00207                                 vol[i] += R->Period[i];
00208                         }
00209                         if (R->Output[i]) vol[i] -= R->Count[i];
00210                 }
00211 
00212                 left = STEP;
00213                 do
00214                 {
00215                         int nextevent;
00216 
00217 
00218                         if (R->Count[3] < left) nextevent = R->Count[3];
00219                         else nextevent = left;
00220 
00221                         if (R->Output[3]) vol[3] += R->Count[3];
00222                         R->Count[3] -= nextevent;
00223                         if (R->Count[3] <= 0)
00224                         {
00225                                 if (R->RNG & 1) R->RNG ^= R->NoiseFB;
00226                                 R->RNG >>= 1;
00227                                 R->Output[3] = R->RNG & 1;
00228                                 R->Count[3] += R->Period[3];
00229                                 if (R->Output[3]) vol[3] += R->Period[3];
00230                         }
00231                         if (R->Output[3]) vol[3] -= R->Count[3];
00232 
00233                         left -= nextevent;
00234                 } while (left > 0);
00235 
00236                 out = vol[0] * R->Volume[0] + vol[1] * R->Volume[1] +
00237                                 vol[2] * R->Volume[2] + vol[3] * R->Volume[3];
00238 
00239                 if (out > MAX_OUTPUT * STEP) out = MAX_OUTPUT * STEP;
00240 
00241                 *(buffer++) = (Bit16s)(out / STEP);
00242 
00243                 count--;
00244         }
00245 }

00280 static void SN76496_set_clock(struct SN76496 *R, int clock) {
00281         /* the base clock for the tone generators is the chip clock divided by 16; */
00282         /* for the noise generator, it is clock / 256. */
00283         /* Here we calculate the number of steps which happen during one sample */
00284         /* at the given sample rate. No. of events = sample rate / (clock/16). */
00285         /* STEP is a multiplier used to turn the fraction into a fixed point */
00286         /* number. */
00287         R->UpdateStep = (unsigned int)(((double)STEP * R->SampleRate * 16) / clock);
00288 }

00291 static void SN76496_set_gain(struct SN76496 *R, int gain) {
00292         int i;
00293         double out;
00294 
00295         gain &= 0xff;
00296 
00297         /* increase max output basing on gain (0.2 dB per step) */
00298         out = MAX_OUTPUT / 3;
00299         while (gain-- > 0)
00300                 out *= 1.023292992;     /* = (10 ^ (0.2/20)) */
00301 
00302         /* build volume table (2dB per step) */
00303         for (i = 0;i < 15;i++)
00304         {
00305                 /* limit volume to avoid clipping */
00306                 if (out > MAX_OUTPUT / 3) R->VolTable[i] = MAX_OUTPUT / 3;
00307                 else R->VolTable[i] = (int)out;
00308 
00309                 out /= 1.258925412;     /* = 10 ^ (2/20) = 2dB */
00310         }
00311         R->VolTable[15] = 0;
00312 }
