00187     /* assume: prefetch buffer hit.

00204 /* this implementation follows what I think the Intel 80386/80486 is more likely
00205  * to do when fetching from prefetch and refilling prefetch --J.C. */
00206 template <class T> static inline T Fetch(void) {
00207     T temp;
00208 
00209     if (prefetch_hit<T>(core.cseip)) {
00210         /* as long as prefetch hits are occurring, keep loading more! */
00211         if ((pq_fill - pq_start) < pq_limit) {
00212             prefetch_filldword();
00213             if (sizeof(T) >= 4 && (pq_fill - pq_start) < pq_limit)
00214                 prefetch_filldword();
00215         }
00216         else {
00217             prefetch_lazyflush(core.cseip + 4 + sizeof(T));
00218         }
00219 
00220         temp = prefetch_read<T>(core.cseip);
00221 #ifdef PREFETCH_DEBUG
00222         pq_hit++;
00223 #endif
00224     }
00225     else {
00226         prefetch_init(core.cseip & (~0x3)); /* fill prefetch starting on DWORD boundary */
00227         prefetch_refill(pq_start + pq_reload); /* perhaps in the time it takes for a prefetch miss the 80486 can load two DWORDs */
00228         temp = prefetch_read<T>(core.cseip);
00229 #ifdef PREFETCH_DEBUG
00230         pq_miss++;
00231 #endif
00232     }
00233 
00234 #ifdef PREFETCH_DEBUG
00235     if (pq_valid) {
00236         assert(core.cseip >= pq_start && (core.cseip+sizeof(T)) <= pq_fill);
00237         assert(pq_fill >= pq_start && (pq_fill - pq_start) <= pq_limit);
00238     }
00239 #endif
00240 
00241     core.cseip += sizeof(T);
00242     return temp;
00243 }
