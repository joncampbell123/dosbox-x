01598                         // user a chance to compensate for it.

01826 void VGA_CheckScanLength(void) {
01827         switch (vga.mode) {
01828         case M_EGA:
01829         case M_LIN4:
01830                 if ((machine==MCH_EGA)&&(vga.crtc.mode_control&0x8))
01831                         vga.draw.address_add=vga.config.scan_len*16;
01832                 else
01833                         vga.draw.address_add=vga.config.scan_len*8;
01834                 break;
01835         case M_VGA:
01836         case M_LIN8:
01837         case M_LIN15:
01838         case M_LIN16:
01839         case M_LIN24:
01840         case M_LIN32:
01841                 if (vga.mode == M_VGA) {
01842                         if (svgaCard == SVGA_TsengET3K || svgaCard == SVGA_TsengET4K) {
01843                                 /* Observed ET4000AX behavior:
01844                                  *    byte mode OR dword mode scans 256-color 4-pixel groups byte by byte from
01845                                  *    planar RAM. word mode scans every other byte (skips by two).
01846                                  *    We can just scan the buffer normally as linear because of the address
01847                                  *    translation carried out by the ET4000 in chained mode:
01848                                  *
01849                                  *    plane = (addr & 3)   addr = (addr >> 2)
01850                                  *
01851                                  *    TODO: Validate that this is correct. */
01852                                 vga.draw.address_add=vga.config.scan_len*((vga.config.addr_shift == 1)?16:8);
01853                         }
01854                         else {
01855                                 /* Most (almost ALL) VGA clones render chained modes as 4 8-bit planes one DWORD apart.
01856                                  * They all act as if writes to chained VGA memory are translated as:
01857                                  * addr = ((addr & ~3) << 2) + (addr & 3) */
01858                                 vga.draw.address_add=vga.config.scan_len*((2*4)<<vga.config.addr_shift);
01859                         }
01860                 }
01861                 else {
01862                         /* the rest (SVGA modes) can be rendered with sanity */
01863                         vga.draw.address_add=vga.config.scan_len*(2<<vga.config.addr_shift);
01864                 }
01865                 break;
01866     case M_PC98:
01867         vga.draw.address_add=vga.draw.blocks;
01868         break;
01869     case M_TEXT:
01870         case M_CGA2:
01871         case M_CGA4:
01872         case M_CGA16:
01873         case M_AMSTRAD: // Next line.
01874                 if (IS_EGAVGA_ARCH || IS_PC98_ARCH)
01875                         vga.draw.address_add=vga.config.scan_len*(2<<vga.config.addr_shift);
01876                 else
01877                         vga.draw.address_add=vga.draw.blocks;
